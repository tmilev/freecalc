\begin{frame}
\begin{theorem}
A line that is perpendicular to a line of slope $k$ has slope $-\frac{1}{k}$.
\end{theorem}
Let $f$ be a function differentiable at a point $a$, and let $m=f'(a)$.

\begin{definition}
\begin{itemize}
\item The line with equation $\displaystyle y-f(a)=m(x-a) $ is called the tangent line to the graph of $f$ at the point $a$.
\item The line with equation $\displaystyle -m(y-f(a))=x-a$ is called the normal line to the graph of $f$ at the point $a$. 
\item If $m\neq 0$, the normal line is also described by the equation $\displaystyle y-f(a)=-\frac{1}{m}(x-a)$.

\end{itemize}
\end{definition}
