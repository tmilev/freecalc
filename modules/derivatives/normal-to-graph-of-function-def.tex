\begin{definition}[Tangent, normal]
\begin{itemize}
\item \textbf{\alertNoH{2}{Tangent.}} \uncover<2->{The line with equation 

\hfil\hfil$\displaystyle \alertNoH{11}{ y-\alertNoH{3}{f(a)}= \alertNoH{4, 12}{m}(x-\alertNoH{5}{a})} $ 

is called the tangent line to the graph of $f$ at the point $a$.
}
\item \textbf{\alertNoH{6}{Normal.}} \uncover<6->{The line with equation 

\hfil \hfil $\displaystyle
\begin{array}{rcll|l}
\displaystyle \alertNoH{10}{-\alertNoH{9}{ m}}(y- \alertNoH{7}{f(a)}) &=& \displaystyle x - \alertNoH{ 8}{a}\\
 \uncover<10->{\displaystyle \alertNoH{11}{y-f(a)}&\alertNoH{11}{{=}}&\displaystyle \alertNoH{11}{\alertNoH{10,13}{ -\frac{1}{m}} (x-a)} &&\alertNoH{10}{ m\neq 0} }
 \end{array}
 $ 

is called the normal line to the graph of $f$ at the point $a$. }
\end{itemize}
\end{definition}
\end{columns}
\uncover<11->{
The normal is perpendicular to the tangent because of the following.
\begin{theorem}
A line that is perpendicular to a \alertNoH{12}{ line of slope $\displaystyle m$} has slope $\displaystyle \alertNoH{13}{-\frac{1}{m}}$.
\end{theorem}
}
