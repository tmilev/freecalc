$
\renewcommand{\arraystretch}{1.9}
\begin{array}{rcl}
\displaystyle \uncover<3->{ \alertNoH{8}{\frac{\diff \alertNoH{4}{y}}{\diff x}}&=&\displaystyle \alertNoH{5}{ \frac{\diff }{\diff x}\left(\alertNoH{4}{\ln (3x+1)} \right)}}\\
