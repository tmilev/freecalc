%\frametitle{Geometric interpretation of integer division}
\vskip -0.1cm
\begin{itemize}
\item Recall integer division of $p$ by $d$ with remainder: $p = q\cdot d + r$.
\item<2-> Integer division of $p$ by $d$ answers the following question:
\end{itemize}
\vskip -0.1cm
\uncover<2->{
\begin{question}
How many times does the length $d$ fit inside the length $p$?
\end{question}
}
\vskip -0.1cm
\begin{itemize}
\item<12-> $p$ may be divisible by $d$ an exact number of times. 
\item $p$ may fail to be integer-divisible by $d$, then we have remainder.
\end{itemize}
\vskip -0.1cm
