\begin{frame}
\begin{observation}
For any number $a$, the following equality holds. 
\[
\alertNoH{5}{\frac{a}{1}} =\alertNoH{6}{ a}
\]
\end{observation}
\uncover<2->{
\begin{example}
Simplify.

\hfil\hfil $
\renewcommand{\arraystretch}{2}
\begin{array}{rcl}
\displaystyle \alertNoH{2,3}{\frac{5}{1}} &\alertNoH{2,3}{=}& \fcAnswer{3}{5}\\
\displaystyle \alertNoH{2,3}{\frac{-3}{1}}&\alertNoH{2,3}{=}&\fcAnswer{3}{-3}\\
\end{array}
$
\end{example}
}
\begin{itemize}
\item<4-> Division by one corresponds to ``division of a segment into one equal part''. 
\item<5-> ``\alertNoH{5}{Division of a segment into one equal part}'' is interpreted as \alertNoH{6}{not dividing the segment at all}.

\end{itemize}
\end{frame}
