\begin{frame}
\begin{definition}[Division]
To divide $p$ by $d$ means to find a number $q$ so that 

\[
p = q \cdot d 
\]
\end{definition}
\begin{definition}[Division with remainder]
To \textbf{divide} a positive integer $p$ by a positive integer $d$  \textbf{with remainder} $r$ means to find the \textbf{largest integer} $q$ and the smallest non-negative remainder $r$ so that:
\[
p = q\cdot d+ r
\]
The number $p$ is called the \textxbf{dividend} and $d$ is called the \textbf{divisor} $q$ is called the \textbf{quotient} and $r$ is called the remainder.
\end{definition}

\end{frame}
