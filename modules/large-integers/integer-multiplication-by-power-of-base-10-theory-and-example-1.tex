\begin{observation}
To multiply numbers that end with zeroes:
\begin{itemize}
\item<11->{\alertNoH{11}{ multiply the non-zero parts ignoring all zeroes at the end;}}
\item<12->{\alertNoH{12}{copy all ending zeroes ignored in the previous step}.}
\end{itemize}
\end{observation}
