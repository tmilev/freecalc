\frametitle{Sum with a negative}
\begin{mathematicalRule}
Subtracting a number is the same as adding its negative.

\[
\begin{array}{rcl}
a+ (-b) &=& a-b\\
-a +b &=& b-a
\end{array}
\]
\end{mathematicalRule}
\begin{example}
\hfil\hfil $\begin{array}{rclcl} 
5+(-3)&=& 5-3 &=& 2\\
10+(-5)&=&10-5 &=& 5\\
-1 +8 &=&8-1&=&7\\
-8+14&=&14-8&=&6
\end{array}$
\end{example}
