\frametitle{Negative of a sum}
\begin{mathematicalRule}
The sum of negatives is the negative of the sum.
\[
- a - b = - (a+b)
\] 
The difference of two numbers is minus the opposite difference.
\[
a-b=-(b-a)
\]
\end{mathematicalRule}
\begin{example}
\hfil\hfil$
\begin{array}{rclclcl}
-5-7&=& -(5+7)&=&-12\\
-7+(-8)&=&-7-8&=&-(7+8)&=&-15\\
5-9&=&- (9-5)&=&-4\\
6-11&=&-(11-6)&=&-5\\
-9+3&=&3-9&=&-(9-3)&=&-6\\
\end{array}
$
\end{example}
