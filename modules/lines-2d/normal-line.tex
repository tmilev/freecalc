\begin{itemize}
\item<3-> Slopes don't change under translation $\Rightarrow$ choose line to pass through origin $O$.
\item<4-> \alertNoH{4}{Let $P$ - point on $L$} \alertNoH{6}{with $x$-coordinate $1$.}
\item<7-> \alertNoH{7,8}{Slope of $L$:} $\alertNoH{7,8}{\alertNoH{12}{m}= \frac{\fcAnswer{8}{\alertNoH{9}{\text{rise}}}}{\fcAnswer{8}{\alertNoH{10}{ \text{run}}} }}\uncover<9->{ = \alertNoH{11}{\frac{\alertNoH{9}{y}}{\alertNoH{10}{1} }}}\uncover<11->{ \alertNoH{11,12}{ =y} .}$
\item<12-> $\Rightarrow$ coordinates of $P$ are $\alertNoH{12}{(1,m)}$.
\item<13-> Rotate by $90^\circ$ the line $L$ around origin $O$ to get line $M$. \uncover<14->{\alertNoH{14}{Let the image of $P$ be $P'$.} }
\item<15-> $\Rightarrow$ \alertNoH{15-19}{ coordinates of $P'$ are $(\fcAnswerUncover{15}{19}{ \alertNoH{20}{-m}}, \fcAnswerUncover{15}{19}{\alertNoH{21}{1}})$}.
\item<22-> $\Rightarrow$ slope of line $M$ is $\frac{\alertNoH{23}{\text{rise}}}{\alertNoH{24}{\text{run}}}=\frac{\alertNoH{23}{1}}{\alertNoH{24}{-m}}$.
\end{itemize}
\end{columns}
\uncover<25->{
\begin{theorem}
A line that is perpendicular to a line of slope $m$ has slope $-\frac{1}{m}$.
\end{theorem}
