$
\begin{array}{@{}r@{~}c@{~}l}

\uncover<5->{ &=&\displaystyle \fcAnswer{6}{ 1} \cdot e^{-x^2}+(x+1) \fcAnswerUncover{5}{8}{e^{-x^2} \alertNoH{9,10}{\left(-x^2\right)'}}}\\

}
\uncover<11->{&\alertNoH{15,16}{{=}} &\displaystyle \alertNoH{15,16}{ \left(\alertNoH{12}{1+ (\alertNoH{13}{x} +\alertNoH{14}{1} )(\alertNoH{13, 14}{ -2x} )} \right) \alertNoH{11}{ e^{-x^2}} \uncover<13->{= \alertNoH{17}{{ \left(\!\! \alertNoH{13}{-2x^2} \alertNoH{14}{-2x} +1\right) e^{-x^2 }}}}}}
