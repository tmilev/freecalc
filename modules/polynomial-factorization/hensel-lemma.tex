\begin{lemma}
Let $h\in \mathbb Z[x]$ with leading coefficient $C$. Let $p$-prime, $p$ does not divide $C$. Let $a_1, a_2\in \mathbb Z_{p}[x]$ be co-prime monic polynomials for which 

\hfil\hfil $h\equiv C a_1 a_2 \mod p$

 $\Rightarrow$ there exist unique monic polynomials $b_1, b_2\in \mathbb Z_{p^{n}}[x]$ such that

\hfil\hfil$
\begin{array}{rcll}
h &\equiv& C b_1 b_2 &\mod p^n \\
b_1 &\equiv& a_1 &\mod p\\
b_2&\equiv& a_2 &\mod p
\end{array}
$
\end{lemma}
\tiny
$a_1, a_2$ - coprime $\stackrel{\text{Bezout's lemma}}{ \Rightarrow}$  $\exists r, s\in \mathbb Z_p[x]$ with $ra_1+ sa_2\equiv 1 \mod p$.

\only<1,2|handout:1,2>{
\begin{proof}[Existence]
Suppose by induction we've found $b_1, b_2$ with $h=Cb_1b_2\mod p^n$, $b_1\equiv a_1\mod p$, $b_2\equiv a_2 \mod p$. Let:

\hfil\hfil $h \equiv  Cb_1b_2 + p^n g \mod p^{n+1},$ 

with $\deg g < \deg h$. Let $r,s$ be the coefficient's from B\'ezout's identity. Divide $sg$ by $a_1$ and $rg$ by $a_2$ over $\mathbb Z_p[x]$ with remainder:

$
\begin{array}{rcl|l}
sg &=& a_1c_1 +d_1 &\deg d_1 < \deg a_1\\
rg &=& a_2c_2 +d_2 &\deg d_2 < \deg a_2\\
\end{array}
$


Multiply the first equality by $a_2$, the second by $a_1$ and add:



\hfil\hfil$
\begin{array}{rcll}
g=1\cdot g \equiv (sa_2+ra_1)g \only<1|handout:1>{
	&\equiv& 
 a_2\left(a_1c_1+d_1\right) + a_1\left(a_2c_2+d_2\right) &\mod p\\
&\equiv& \left(a_2d_1+a_1d_2\right) + a_1a_2\left(\underbrace{c_1+c_2}_{\equiv 0 \mod p, \text{ see below}}\right)&\mod p
 \\}
&\equiv& a_2d_1+a_1d_2 ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ &\mod p
\end{array}
$

\only<1|handout:1>{
$\deg (LHS) = \deg g<\deg h = \deg a_1 + \deg a_2$. Since $a_2d_1+a_1d_2<\deg h$, only way to have $\deg (RHS)=\deg (LHS)$ is to have that $\deg a_1a_2(c_1+c_2) < \deg h$ $\Rightarrow c_1+c_2\equiv 0 \mod p$.
}

\only<2|handout:2>{

We claim
$
b_1 + p^n C^{-1}\cdot d_1$, $
b_2+ p^n C^{-1}\cdot d_2$ satisfy the lemma for $n+1$. Indeed, compute:




$
\begin{array}{rcll}
C(b_1+p^nC^{-1}d_1)(b_2+p^nC^{-1}d_2)&\equiv& C\left(b_1b_2 + p^nC^{-1}\left(b_1d_2 + b_2 d_1\right)+ p^{2n}C^{-2}d_1d_2 \right) &\mod p^{n+1}\\
&\equiv&Cb_1b_2 + p^nC\cdot C^{-1}\left(a_1d_2 +a_2d_1\right)&\mod p^{n+1}\\
&\equiv&Cb_1b_2+pg&\mod p^{n+1}
\end{array}
$
}
\end{proof}
}

\begin{lemma}[Hensel lift]
Let $h\in \mathbb Z[x]$, leading coefficient $C$, $p$-prime, $p$ does not divide $C$, let $a,b\in \mathbb Z_{p}[x]$ be co-prime, monic such that:
		
\hfil\hfil $\alertNoH{2,24}{h\equiv C a b \mod p}$
		
$\Rightarrow$ there exist unique monic polynomials $a_n, b_n\in \mathbb Z_{p^{n}}[x]$ such that:
		
\hfil\hfil$
\begin{array}{rcll}
\alertNoH{2,9}{h} &\alertNoH{2,9}{\equiv}& \alertNoH{2,9}{C a_n b_n} &\alertNoH{2,9}{\mod p^n} \\
\alertNoH{2,23}{a_n} &\alertNoH{2,23}{\equiv}& \alertNoH{2,23}{a} &\alertNoH{2,23}{\mod p}\\
\alertNoH{2,23}{b_n}&\alertNoH{2,23}{\equiv}&\alertNoH{2,23}{ b} &\alertNoH{2,23}{\mod p}
\end{array}
$
\end{lemma}
\tiny
	\vskip -0.1cm
\begin{proof}
	\uncover<4->{
\alertNoH{4,6,23}{Suppose by induction we've proved the lemma for $n-1$: }} \uncover<5->{we have unique $a_{n-1}, b_{n-1}$ with 

\hfil\hfil$\begin{array}{rcll}
\alertNoH{5,7}{h}&\alertNoH{5,7}{\equiv}& \alertNoH{5,7}{Ca_{\alertNoH{6}{n-1}}b_{\alertNoH{6}{n-1}}}  ~~~~~~~~~~~~~~~~~~~~~~~~~ & \alertNoH{5,7}{\mod p^{\alertNoH{6}{n-1}}}\\
\uncover<7->{\alertNoH{7,8,21}{ h-Ca_{n-1}b_{n-1}}&\alertNoH{7,8,21}{=}& \fcAnswer{8}{\alertNoH{21}{\alertNoH{22}{ p^{n-1}} g}}   &\alertNoH{7,8,21,22}{\mod p^n} }
\end{array}
$ 
}

\uncover<8->{for some $g$ with $\deg g < \deg h$.} \uncover<9->{ The solutions for $p^{n}$ are also solutions for $p^{n-1}$ and so by the uniqueness for $n-1$:}

\uncover<9->{
\hfil\hfil$ 
\begin{array}{rcll} 
\alertNoH{9,10}{a_{n}} &\alertNoH{9,10}{=}& \alertNoH{9,10}{a_{n-1} + p^{n-1} A} &\alertNoH{9}{ \mod p^n}\\
\alertNoH{9,10}{b_{n}} &\alertNoH{9,10}{=}& \alertNoH{9,10}{b_{n-1} + p^{n-1} B}&\alertNoH{9}{\mod p^n}
\end{array}
$
}

\uncover<3->{Solve for $a_n, b_n$:}

$
\begin{array}{rcl|l} 
\uncover<3->{\alertNoH{3}{h}&\alertNoH{3}{\equiv}&\alertNoH{3}{C \alertNoH{10}{a_n} \cdot \alertNoH{10}{b_n}} &\alertNoH{3}{\mod p^n}}\\
\uncover<10->{
h&\equiv &\alertNoH{11}{\alertNoH{12,13,14,17}{ C} \left(\alertNoH{10}{ \alertNoH{12,15}{a_{n-1}} +\alertNoH{13,17}{\alertNoH{14}{ p^{n-1}} \alertNoH{16}{A}} }\right) \left(\alertNoH{10}{ \alertNoH{12,16}{b_{n-1}} + \alertNoH{13,17}{ \alertNoH{14}{ p^{n-1}} \alertNoH{15}{B}} }\right)} &\mod p^{n}}\\
\uncover<11->{\alertNoH{20}{h} &\equiv& \alertNoH{11}{\alertNoH{12,20}{Ca_{n-1}b_{n-1}} + \alertNoH{13,19}{ \alertNoH{14}{Cp^{n-1}} \left( \alertNoH{15}{a_{n-1}B} + \alertNoH{16}{b_{n-1}A}\right)} + \alertNoH{17,18}{Cp^{2n-2}AB}}& \alertNoH{18}{ \mod p^n}} \\
\uncover<18->{ \alertNoH{19}{Cp^{n-1}\left( a_{n-1}B + b_{n-1}A\right) }&\equiv& \alertNoH{20,21}{h-Ca_{n-1}b_{n-1}}& \mod p^n} \\
\uncover<21->{C\alertNoH{22}{ p^{n-1}}\left( a_{n-1}B + b_{n-1}A\right)&\equiv& \alertNoH{21} {\alertNoH{22}{p^{n-1}}g}& \alertNoH{22}{\mod p^n}}\\
\uncover<22->{C\left( \alertNoH{23}{a_{n-1}}B + \alertNoH{23}{b_{n-1}}A\right)&\equiv& g& \alertNoH{22,23}{\mod p}}\\
\uncover<23->{\alertNoH{26}{C\left( \alertNoH{23,24}{a} \alertNoH{25}{B} + \alertNoH{23,24}{b} \alertNoH{25}{A} \right)}&\alertNoH{26}{\equiv}&\alertNoH{26}{ g}& \alertNoH{23,26}{\mod p}}
\end{array}$

\uncover<24->{$\alertNoH{24}{a,b}$ are given $\Rightarrow$ \uncover<25->{ that's a linear equation $\mod p$ for the coefficients of $\alertNoH{25}{A,B}$.}}
\end{proof}
