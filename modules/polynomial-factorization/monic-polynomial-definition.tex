\begin{frame}
\begin{definition}A polynomial $p$ is called monic if its leading coefficient is $1$.
\end{definition}

\begin{example}
\begin{itemize}
\item $x^2+1$ is monic.
\item $-x^2+1$ is not.
\item $2x$ is not.

\item $(x^2+1) \mod 3$ is monic.
\item $(2x^2+1) \mod 3$ is not.
\item $4x^2+1 = (x^2+1) \mod 3$ is monic.
\end{itemize}
\end{example}
\end{frame}
