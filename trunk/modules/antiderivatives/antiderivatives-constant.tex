\begin{frame}
\begin{theorem}
If $F$ is an antiderivative of $f$ on an interval $I$, then an arbitrary antiderivative of $f$ on $I$ is of the form
\[
F(x) + C
\]
where $C$ is an arbitrary constant.
\end{theorem}
\end{frame}
