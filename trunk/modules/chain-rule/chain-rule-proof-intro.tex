\begin{frame}
\begin{theorem}[Chain rule]
Let \alertNoH{6,11,12}{$g$-differentiable at neighborhood of $a$}, \alertNoH{9,13}{$f$-diff. at neighb. of $g(a)$}.
\[
\alertNoH{14}{  \alertNoH{3}{\left(f(g(x))\right)'_{|x=a}}= f'(g(a)) g'(a)}
\]
\end{theorem}
\begin{proof}[Proof with \alertNoH{2}{additional assumptions} -motivation for actual proof]
\uncover<2->{
\alertNoH{2}{Suppose that \alertNoH{5}{$g(x)\neq g(a)$ so long as $x\neq a$}}.  \uncover<8->{\alertNoH{8}{Set $G(y)=\frac{f(y)-f(g(a))}{y-g(a)}$.}} \uncover<9->{\alertNoH{9}{ $G(y)$ is continuous at $g(a)$} $\Rightarrow$} \uncover<10->{\alertNoH{10}{$G(g(x))$ is continuous at $a$.}} \uncover<12->{\alertNoH{12}{Furthermore $g(x)$ is continuous at $a$.}}

$\begin{array}{rcl}
\uncover<3->{\alertNoH{14}{ \alertNoH{3}{(f \circ g)'(a)}} &=& \uncover<4->{\displaystyle\lim_{x\to a} \frac{f(g(x)) - f(g(a)) }{x-a}} }\\
\uncover<5->{&=&\displaystyle \alertNoH{6,7}{ \lim_{x\to a} } \alertNoH{7}{\left(\frac{f(g(x)) - f(g(a)) }{\alertNoH{5}{g(x)-g(a)}}\right)} \alertNoH{6}{ \left(\frac{ \alertNoH{5}{ g(x)-g(a)} }{x-a}\right)} }\\
\uncover<6->{&=&\displaystyle \alertNoH{7-10}{\lim_{x\to a}\left( \frac{f(\alertNoH{12}{g(x)}) - f(g(a)) }{\alertNoH{12}{g(x)}-g(a)}\right)}  \alertNoH{6,11}{\lim_{x\to a}\left(\frac{g(x)-g(a)}{x-a}\right)} }\\
\uncover<11->{&=&\displaystyle \alertNoH{13}{\left(\lim_{\alertNoH{12}{y=g(x),y\to g(a)}} \frac{f(\alertNoH{12}{y}) - f(g(a)) }{\alertNoH{12}{y}-g(a)}\right) } \alertNoH{11}{g'(a)}=\uncover<13->{\alertNoH{14}{ \alertNoH{13}{f'(g(a))}g'(a)}}\quad .}
\end{array}
$
}
\end{proof}
