\solution{\ref{d/dxln((3x+1)/(4x-5))}

$\begin{array}{rcll|l}
\displaystyle \frac{\diff}{\diff x}\left(\ln \left(\frac{3x+1}{4x-5}\right)\right)&=&\displaystyle  \frac{\diff}{\diff x}\left(\ln (3x+1)-\ln(4x-5)\right)\\
&=&\displaystyle \frac{(3x+1)'}{3x+1}-\frac{(4x-5)'}{4x-5}\\
&=&\displaystyle \frac{3}{3x+1}-\frac{4}{4x-5}\\
&=&\displaystyle \displaystyle \frac{-19}{12 x^{2}-11 x-5} &&\text{step optional.}
\end{array}
$
