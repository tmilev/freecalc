\pointsii{1}  $y = \sqrt[3]{4+3\tan x}$
\ans{%
\begin{align*}
\text{Let } \quad u & = 4+3\tan x. \\
\text{Then } \quad y & = u^{1/3}. \\
\text{Chain Rule: } \quad \frac{\diff y}{\diff x} & = \frac{\diff y}{\diff u}\frac{\diff u}{\diff x} \\
 & = \big(\frac{1}{3}u^{-2/3}\big) (3\sec^2x) \\
 & = \frac{1}{3} \sec^2x (4+3\tan x)^{-2/3}.
\end{align*}
}%
