\pointsii{5}  $y = \sqrt{\sec 4x}$
\solution{%
\begin{align*}
\text{Chain Rule: } \quad \frac{\diff y}{\diff x} & = \big( \frac{1}{2}(\sec 4x)^{-1/2}\big) \frac{\diff}{\diff x}(\sec 4x) \\
\text{Chain Rule: } \quad \frac{\diff y}{\diff x} & = \Big( \frac{1}{2\sqrt{\sec 4x}}\Big) (\sec 4x \tan 4x)\frac{\diff}{\diff x}(4x) \\
 & = \Big( \frac{1}{2\sqrt{\sec 4x}}\Big) (\sec 4x \tan 4x)(4) \\
 & =  \frac{2\sec 4x\tan 4x}{\sqrt{\sec 4x}} \\
\intertext{There are many ways to simplify this answer, including both of the following.}
 & =  2\sqrt{\sec 4x}\tan 4x. \\
 & =  2(\sec 4x)^{3/2}\sin 4x. 
\end{align*}

\pointsii{5}  $y = \frac{1+\sin x^2}{1+\cos x^2}$
\solution{%
\begin{align*}
\text{Quotient Rule: } \quad \frac{\diff y}{\diff x} & = \frac{(1+\cos x^2)\frac{\diff}{\diff y}(1+\sin x^2) - (1+\sin x^2)\frac{\diff}{\diff x}(1+\cos x^2)}{(1+\cos x^2)^2} \\
\intertext{By the Chain Rule, $\frac{\diff}{\diff x}(1+\cos x^2) = -2x\sin x^2$ and $\frac{\diff}{\diff x}(1+\sin x^2) = 2x\cos x^2$.}
\frac{\diff y}{\diff x} & = \frac{(1+\cos x^2)(2x\cos x^2) - (1+\sin x^2)(-2x\sin x^2)}{(1+\cos x^2)^2} \\
 & = \frac{2x\cos x^2 + 2x\cos^2x^2 + 2x\sin x^2 + 2x\sin^2x^2}{(1+\cos x^2)^2} \\
 & = \frac{2x(\cos^2 x^2 + \sin^2 x^2) + 2x(\cos x^2 + \sin x^2)}{(1+\cos x^2)^2} \\
\intertext{By the Pythagorean Identity, $\cos^2x^2 + \sin^2x^2 = 1$.}
\frac{\diff y}{\diff x} & = \frac{2x + 2x(\cos x^2 + \sin x^2)}{(1+\cos x^2)^2} \\
 & = \frac{2x(1 + \cos x^2 + \sin x^2)}{(1+\cos x^2)^2}.
\end{align*}
}%
