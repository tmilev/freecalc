\begin{theorem}[Conjugation preserves $+,\cdot$]
\begin{enumerate}
\item $\alertNoH{18}{\overline {z\cdot w}=\overline z\cdot \overline w}$
\item $\alertNoH{25}{\overline {z+w}=\overline z+\overline w}$
\end{enumerate}
\end{theorem}
\uncover<2->{
\begin{proof}
Let $\alertNoH{4,11}{z=x+yi}$, $\alertNoH{5,12}{w=u+vi}$.
\begin{itemize}
\item<3-> $\begin{array}{rcl}
\alertNoH{18}{\overline {\alertNoH{4}{z}}\cdot \overline{\alertNoH{5}{w}}}  &=& \uncover<4->{ \overline{(\alertNoH{4}{x+yi})}\cdot \overline{(\alertNoH{5}{u+vi})} } \uncover<6->{=  (\alertNoH{7,9}{x} \alertNoH{8,10}{-yi})( \alertNoH{7,10}{u}\alertNoH{8,9}{-vi})}\\
\uncover<7->{&=& \alertNoH{18}{(\alertNoH{7}{xu}\alertNoH{8}{-yv})\alertNoH{9,10}{-}( \alertNoH{9}{xv} +\alertNoH{10}{yu} )\alertNoH{9,10}{ i}}} \\
\alertNoH{18}{\overline{\alertNoH{11}{z} \cdot \alertNoH{12}{w}}} &=& \uncover<11->{ \overline{ (\alertNoH{11}{\alertNoH{13,15}{x} +\alertNoH{14,16}{yi}})\alertNoH{12}{ ( \alertNoH{13,16}{u} +\alertNoH{14,15}{iv})}}} \uncover<13->{ = \overline{ \alertNoH{13}{xu} \alertNoH{14}{-yv} + (\alertNoH{15}{xv}+\alertNoH{16}{yu})\alertNoH{15,16}{i}}} \\

\end{array}
$
\item<19-> $\begin{array}{rcl}
\alertNoH{25}{\overline {z}+\overline{w}} &=&\uncover<19->{ \overline{(x+yi)}+\overline{(u+vi)}} \uncover<20->{=(x-yi)+(u-vi)}\\

\end{array}
$
\end{itemize}
\end{proof}

In the preceding slide we proved the following.
\begin{theorem}[Conjugation preserves $\cdot$]
$\overline {z\cdot w}=\overline z\cdot \overline w$.
\end{theorem}
\begin{corollary}
\alertNoH{3}{$|zw|=|z||w|$}.
\end{corollary}
\begin{proof}
$\uncover<2->{|zw|= \sqrt{zw \alertNoH{3}{\overline{zw}}}} \uncover<3->{= \sqrt{zw \alertNoH{3}{\bar z \bar w}} } \uncover<4->{ = \sqrt{z\bar z}\sqrt{w\bar w}} \uncover<5->{ = |z| |w|} $.
\end{proof}
\uncover<6->{
\begin{corollary}
$|\frac{z}{w}|=\frac{|z|}{|w|}$, $w\neq 0$.
\end{corollary}
}
