\begin{itemize}
\item Let $z,w\neq 0 $ \uncover<2->{and let

\hfil $\alertNoH{12}{ \alertNoH{3}{ |z| =e^\rho},\quad \alertNoH{4}{ |w|=e^\sigma}}.$
}
\item<5-> Let $\alpha,\beta$ be arguments of $z,w$.
$
\begin{array}{rcl}
z&=& e^{\rho}(\cos \alert<handout:0|6>{\alpha} + i \sin \alert<handout:0|6>{\alpha} )=e^{\rho +i\alpha} \\
w&=& e^{\sigma}(\cos \alert<handout:0|7>{\beta} +i\sin \alert<handout:0|7>{\beta} )=e^{\sigma +i\beta} .\end{array}
$
\end{itemize}

\uncover<8->{%
\begin{theorem}[Summary]
$\begin{array}{rcl}
\alertNoH{9}{z w} &=& \alertNoH{12}{|z|}( \cos\alpha+ i\sin\alpha) \alertNoH{12}{ |w|} ( \cos \beta +i\sin \beta )\\
&=&e^\rho ( \cos\alpha+ i\sin\alpha) e^\sigma ( \cos \beta +i\sin \beta )= e^{ \rho + i \alpha} e^{\sigma+i\beta} \\
&=&e^{\rho+\sigma +i(\alpha+\beta)} = \alertNoH{12}{|z||w|} ( \cos (\alertNoH{10}{\alpha+\beta})+i\sin (\alertNoH{10}{\alpha+\beta}) ) .\end{array}$
\end{theorem}
}%

\item<10-> An argument (polar angle) of $zw$ is $\alertNoH{10}{\alpha+\beta} $.
\item<11-> $\Rightarrow$ Multiplying complex numbers adds arguments (polar angles).
\item<12-> Multiplying complex numbers multiplies absolute values.
