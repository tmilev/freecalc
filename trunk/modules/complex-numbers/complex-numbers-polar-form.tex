\begin{lemma}
$\alert<handout:0|1>{\left|\frac{z}{|z|}\right| } \uncover<2->{\alertNoH{2}{=\frac{|z|}{|z|}}} \uncover<3->{ \alertNoH{3}{ =1 }.}$
\end{lemma}

\begin{itemize}
\item<4-> Let $z=x+iy$ be a non-zero complex number.
\item<5-> Then $0$, $z$, $\frac{z}{|z|}$ lie on a ray \uncover<7->{and $\frac{z}{|z|}$ lies on the unit circle.}
\item<8-> Let $\theta$ - angle between the real axis and the ray between $0$ and $z$.
\item<9-> Then $\alertNoH{14}{\frac{z}{|z|}= \alertNoH{12}{ \alertNoH{10}{ \cos \theta} +i \alertNoH{11}{\sin \theta}}} \uncover<12->{ \alertNoH{12}{= e^{i\theta}}} $.
\uncover<13->{
\begin{definition}[Polar form of complex numbers]
Let $z\neq 0$. $\alertNoH{14}{z=|z|(\alertNoH{15}{\cos\theta+i\sin \theta)}}= |z| \alertNoH{15}{ e^{i\theta}} $ is called polar form of $z $.
\end{definition}
}
\item<16-> Let $\alertNoH{19}{\rho = \ln \alertNoH{17}{|z|} } \uncover<17->{ = \ln \alertNoH{17}{\sqrt{ x^2+y^2} }}  \uncover<18->{ = \frac{1}{2}\ln(x^2+y^2)}$.
\item<19-> Then $z=\alertNoH{19}{|z|} ( \cos\theta+ i\sin\theta) = \alertNoH{19}{ e^{\rho}}( \cos \theta +i\sin \theta) \uncover<20->{ =e^{\rho} e^{i\theta } }\uncover<21->{=e^{\rho + i \theta}.} $
\end{itemize}

\begin{definition}[Polar form of complex numbers]
Let $z\neq 0$. Then $z=|z|(\cos \alertNoH{2}{ \theta} +i\sin \alertNoH{2}{\theta} )$ is called polar form of $z $.
\end{definition}
\begin{itemize}
\item<2-> \alertNoH{2}{$\theta$} is called an argument of $z$. We write \[\theta=\arg z.\]
\end{itemize}
\end{columns}

\begin{itemize}
\item<3-> If $\theta$ is an argument of $z$, so is $\theta + 2k\pi$ for all integers $k$.
\item<4-> If $\theta\in (-\pi, \pi]$, we say that $\theta$ is the principal argument of $z$.
\item<5-> If we write $\theta=\arg z$ without clarifying the choice of the argument, it is implied that $\theta$ is the principal argument of $z$, $\theta\in (-\pi,\pi]$.
\item<6-> One should never write $\theta=\arg z$ without clarifying the choice of argument.
