\begin{theorem}[Euler's formula]
$e^{i\theta}=\cos \theta+i\sin \theta$.
\end{theorem} 
\begin{lemma}
$\alert<handout:0|1>{\left|\frac{z}{|z|}\right| } \alert<2->{=\frac{|z|}{|z|}}\alert<3->{=1}$.
\end{lemma}
\end{columns}
\begin{itemize}
\item<4-> Let $z=x+iy$ be a non-zero complex number.
\item<5-> Then $0$, $z$, $\frac{z}{|z|}$ lie on a ray \uncover<7->{and $\frac{z}{|z|}$ lies on the unit circle.}
\item Let $\theta$ - angle between the real axis and the ray between $0$ and $z$.
\item Then $\frac{z}{|z|}= \cos \theta +i\sin \theta$.
\item This gives a geometric proof/motivation for Euler's formula when $\theta $ a real number (Euler's f-la holds over all complex numbers).
\item Let $\rho = \ln |z|= \ln \sqrt{x^2+y^2}=\frac{1}{2}\ln(x^2+y^2)$.
\item Then $z=|z|(\cos\theta+i\sin\theta)= e^{\rho}(\cos\theta+i\sin \theta)$.
\end{itemize}
\begin{definition}
Let $z\neq 0$. Then $z=e^{\rho}(\cos\theta+i\sin \theta)$ is called polar form of $z $.
\end{definition}
