\column{0.65\textwidth}
\[
\lim\limits_{(x,y) \to (0,0)} \frac{xy^2}{x^2+y^4}
\]
\uncover<2->{
Along $y=mx$:
$
\begin{array}{l}
\displaystyle\frac{xy^2}{x^2+y^4} = \uncover<3->{\frac{m^2 x^3}{x^2+m^4x^4}} \uncover<4->{=  \frac{m^2 x}{1+m^2x^4}} \uncover<5->{\alertNoH{12}{ \to 0}} \\
\displaystyle \uncover<5->{\alertNoH{12}{ \text{ as } x \to 0}.}
\end{array}
$
}
\end{columns}
\uncover<6->{For direction $x=0$, $y=m$: directional limit is again $0$.}
\uncover<7->{$\Rightarrow$ all directional limits exist and equal $0$.}
\uncover<8->{However, along $x=y^2$:}
\[
\uncover<8->{\frac{xy^2}{x^2+y^4} =}\uncover<9->{ \frac{y^4}{y^4+y^4} = }\uncover<10->{ \frac{1}{2} }\uncover<11->{\alertNoH{12}{\to  \frac{1}{2} \text{ as } x \to 0}}
\]
\uncover<12->{\alertNoH{12}{Therefore $\lim\limits_{(x,y) \to (0,0)} \frac{xy^2}{x^2+y^4}$ does not exist.}}
\end{example}
