\hfil \hfil $\displaystyle \lim_{(x,y) \to (0,0)} \frac{x^2y}{x^2+y^2}
$
\begin{itemize}
\item<2-> $f$ is not defined at $P_0(0,0)$;
\item<3-> Even if it were, the actual value might be different from the limit.

\end{itemize}
\uncover<4->{Polar coordinates: \alertNoH{9,10}{ $x=r\cos\theta$, $y =r \sin\theta$}. \alertNoH{8}{ We have: \alertNoH{5,6}{$(x,y) \to (0,0)$ if an only if \fcAnswer{6}{$r\to 0$  \alertNoH{9,10}{in polar coordinates}.}}}}
\end{columns}

\[
\begin{array}{rcl}
\uncover<7->{ \displaystyle \lim\limits_{\alertNoH{8}{(x,y) \to (0,0)}} \alertNoH{9,10}{ \frac{x^2y}{x^2+y^2} } &\alertNoH{9,10}{=}& \displaystyle  \lim\limits_{\alertNoH{8}{r\to 0}}   \fcAnswer{10}{\alertNoH{10}{\frac{\fcCancel{11}{ r^2}\cos^2\theta r\sin\theta}{\fcCancel{11}{r^2}}}}} \\
\uncover<11->{&\alertNoH{0}{=}&\displaystyle \lim\limits_{r\to 0}\phantom{\textbf{?}} \alertNoH{12}{r\cos^2{\theta}\sin\theta}}\\
\uncover<15->{&\alertNoH{0}{=}&0}
\end{array}
\]
\uncover<15->{For the last equality, we use the squeeze theorem:}
\uncover<12->{$\uncover<14->{0=} \uncover<13->{\lim\limits_{r\to 0} } -r \leq \uncover<13->{ \lim \limits_{r \to 0}} \alertNoH{12}{ r\cos^2{\theta}\sin\theta } \leq \uncover<13->{\lim\limits_{r\to 0}}  r \uncover<14->{=0.}$}
