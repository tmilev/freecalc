\begin{frame}
\begin{example}
Show that there is a root of the equation
\abovedisplayskip=0pt
\abovedisplayshortskip=0pt
\belowdisplayskip=0pt
\belowdisplayshortskip=0pt
\[
x = \cos x.
\]

\begin{columns}[t]
\column{0.4\textwidth}
\begin{itemize}
\item<2->  Let \alert<2-3>{$f(x) = $ \uncover<3->{$x-\cos x$.}}
\item<4->  $f$ is continuous.  
\item<5->  Use the IVT with \alert<5-6>{$a = $ \uncover<6->{$0$}}, \alert<7-8>{$b = $ \uncover<8->{$\pi /2$}}, and $N = 0$.
\item<9->  \alert<9-10>{$f(0) = $ \uncover<10->{$-1$.}}
\item<11->  \alert<11-12>{$f(\pi /2) = $ \uncover<12->{$\pi/2$.}}
\item<13->  $f(0) < 0 < f(\pi /2)$.  
\item<14-16>  Therefore there is a $c$ between $0$ and $\pi /2$ such that $f(c) = 0$.
\end{itemize}

\column{0.6\textwidth}

\end{example}
