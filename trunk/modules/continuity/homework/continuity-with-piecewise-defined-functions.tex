Find the (implied) domain of $f(x)$. Extend the definition of $f$ at $x=3$ to make $f$ continuous at $3$.
\begin{multicols}{2}
\begin{enumerate}
\item $f(x)=\frac{x^2-x-6}{x-3}$.

\answer{
\begin{tabular}{l}
Implied domain: $x\in (-\infty, 3)\cup (3,\infty)$. \\
Extend $f(x)$ to $\bar f(x)=x+2$.
\end{tabular}
}
\item $f(x)=\frac{x^3-27}{x^2-9}$.

\answer{
\begin{tabular}{l}
Implied domain: \\
$x\in (-\infty, -3)\cup(-3,3)\cup (3,\infty)$. \\
Extend $f(x)$ to \\
$\bar f(x)=\frac{x^2+3x+9}{x+3}$\\ 
with domain $x\in (-\infty, -3)\cup(-3,\infty)$.
\end{tabular}
}
\end{enumerate}
\end{multicols}
