% begin module inflection-point-def
\begin{frame}
\begin{definition}[Inflection Point]
Let $f$ be a twice differentiable function. A point $P$ on a curve $y = f(x)$ is called an inflection point if $f$ changes from concave up to concave down or from concave down to concave up at $P$.
%note that a curve being concave up/down implies continuity. 
\end{definition}
\uncover<2->{%
Another way of saying this is that $P$ is an inflection point if $f''$ changes signs at $P$.
}%
\end{frame}
