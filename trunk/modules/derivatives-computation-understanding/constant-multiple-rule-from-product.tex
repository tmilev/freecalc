\[
\alert<6>{(cf)'=c f'}

\[
\uncover<2->{\alert<3,6>{(cf)'}=} \uncover<3->{\alert<3>{\alert<4>{(c)'} f+ c f'}=}\uncover<4->{\alert<4>{0} f+cf'=}\uncover<5->{\alert<6>{cf'}}
\]
\uncover<6->{\alert<6>{as desired}.}
\end{example}
