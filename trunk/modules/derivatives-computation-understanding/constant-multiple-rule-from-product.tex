\begin{frame}
\begin{example}
Let $c$ be a constant. Derive the constant multiple rule
\[
\alertNoH{6}{(cf)'=c f'}
\]
\uncover<3->{using the \alertNoH{3}{product rule $(fg)'=f'g+fg'$}} \uncover<4->{\alertNoH{4}{and the constant derivative rule $(c)'=0$.}}

\[
\uncover<2->{\alertNoH{3,6}{(cf)'}=} \uncover<3->{\alertNoH{3}{\alertNoH{4}{(c)'} f+ c f'}=}\uncover<4->{\alertNoH{4}{0} f+cf'=}\uncover<5->{\alertNoH{6}{cf'}}
\]
\uncover<6->{\alertNoH{6}{as desired}.}
\end{example}

\end{frame}
