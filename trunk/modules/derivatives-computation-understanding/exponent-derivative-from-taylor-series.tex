% begin module power-rule-rationa-from-chain-rule
\begin{frame}

\begin{example}
Derive \alertNoH{11}{the exponent rule $\left(e^x\right)'=e^x$} \uncover<2->{using the Calc II formula below,} \uncover<3->{ \alertNoH{3}{the infinite} {\color{gray!50} (both sides uniformly convergent)} \alertNoH{3}{sum rule $(f_1+f_2 +f_3 +\dots)' =f_1' +f_2'  +f_3' +\dots$}}
\uncover<4->{and \alertNoH{4}{the power rule $(x^n)'=nx^{n-1}$}. }
\uncover<2->{\[
\alertNoH{2,10}{e^x}=\alertNoH{2,10}{1+x+\frac{x^2}{2!}+\frac{x^3}{3!}+\dots},
\]
}
\uncover<2->{where $n!=1\cdot2\cdot3\cdot \dots\cdot n$.} \uncover<5->{We have that $\alertNoH{5,9}{\frac{n}{\alertNoH{6}{ n!}}} =\uncover<6->{ \frac{\alertNoH{7}{n} }{ \alertNoH{6}{ 1\cdot 2\cdot \dots\cdot (n-1) \cdot  \alertNoH{7}{n}}} =} \uncover<7->{ \frac{1}{\alertNoH{8}{1\cdot 2\cdot \dots\cdot (n-1)}} =} \uncover<8->{ \alertNoH{9}{ \frac{1}{ \alertNoH{8}{(n-1)!} }}.} $}
\[
\begin{array}{rcl}
\uncover<2->{\alertNoH{11}{\left(\alertNoH{2}{e^x}\right)'} &=& \alertNoH{3}{ \left( \alertNoH{2}{1+x+\frac{x^2}{2!}+\frac{x^3}{3!}+\dots} \right)' }} \\
\uncover<3->{&=& \alertNoH{3}{ \alertNoH{4}{\left(1\right)'} +\alertNoH{4}{ \left( x \right)'}+\frac{\alertNoH{4}{\left(x^2\right)'}}{2!} +\frac{\alertNoH{4}{ \left( x^3\right)'}}{3!}+\dots + \frac{\alertNoH{4}{ \left(x^n\right)'}}{n!}+\dots}}
\\ \uncover<4->{&=&\alertNoH{4}{0}+ \alertNoH{4}{1}+ \frac{\alertNoH{4}{ \alertNoH{5-9}{2}x}}{\alertNoH{5-9}{2!}}+ \frac{\alertNoH{4}{ \alertNoH{5-9}{3} x^2}}{\alertNoH{5-9}{3!}}+ \dots +\frac{\alertNoH{4}{ \alertNoH{5-9}{n}x^{n-1}}}{ \alertNoH{5-9}{n!}}+\dots }\\
\uncover<9->{&=& \alertNoH{10}{ \phantom{ 0~ + }  1 + \frac{\phantom{2}x}{ \alertNoH{9}{1!}}+\frac{ \phantom{3}x^{2}}{\alertNoH{9}{2!}} +\dots+ \frac{ \phantom{n} x^{n-1}}{\alertNoH{9}{(n-1)!}}+ \dots }=}\uncover<10->{ \alertNoH{10, 11}{ e^x}}\\
\end{array}
\]
\uncover<11->{\alertNoH{11}{as desired.}}
\end{example}

\end{frame}
