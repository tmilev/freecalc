\solution{\ref{problemd/dx((tanx)e^x)}
\[
\begin{array}{rcll|l}
\displaystyle \frac{\diff}{\diff x}\left((\tan x)e^x\right)&=&\displaystyle
\left(\frac{\diff}{\diff x}\left(\tan x\right)\right)e^x+ \left(\tan x\right) \frac{\diff}{\diff x}\left(e^x\right)&&\text{product rule}\\
&=&\displaystyle (\sec^2 x) e^x+ (\tan x) e^x\\
&=&\displaystyle (\sec^2 x+\tan x)  e^x.
\end{array}
\]
