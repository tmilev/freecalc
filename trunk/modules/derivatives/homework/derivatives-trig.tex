
(Textbook, page 146, problems 1-16).
Compute the derivative.
\begin{multicols}{2}
\begin{enumerate}
\item $f(x)= 3x^2 -2 \cos x$.

\answer{$ 6 x+2 \sin{}(x)$}
\item $f(x)=\sqrt{x}\sin x$.

\answer{$ x^{\frac{1}{2}}\cos{}(x) +\frac{1}{2}x^{-\frac{1}{2}} \sin{}(x) $}
\item $f(x)=\sin x +\frac{1}{2}\cot x$.

\answer{$\frac{-\frac{1}{2}+\cos{}(x) (\sin{}(x))^{2}}{(\sin{}(x))^{2}} $}
\item $y=2\sec x - \csc x$.

\answer{$ \frac{(\cos{}(x))^{3}+2 (\sin{}(x))^{3}}{(\cos{}(x) \sin{}(x))^{2}}$}
\item $y=\frac{1+(\sin{}(\theta))^{2}}{(\cos{}(\theta))^{3}}$.

\answer{$ \frac{1+(\sin{}(t))^{2}}{(\cos{}(t))^{3}}$}
\item $g(t)=4 \sec t + \tan t$.

\answer{$\frac{1+4 \sin{}(t)}{(\cos{}(t))^{2}} $}
\item $y= c\cos t + t^2\sin t$.

\answer{$ - c \sin{}(t)+2 t \sin{}(t)+ t^{2}\cos{}(t)$}
\item $y=u(a\cos u + b \cot u)$.

\answer{$ \frac{- a u (\sin{}(u))^{3}+a \cos{}(u) (\sin{}(u))^{2}- b u +b \cos{}(u) \sin{}(u)}{(\sin{}(u))^{2}}$}
\item $y=\frac{x}{2-\tan x}$.

\answer{$ \frac{x - \cos{}(x) \sin{}(x)+2 (\cos{}(x))^{2}}{(2 \cos{}(x)- \sin{}(x))^{2}}$}
\item $y=\sin \theta \cos \theta$.

\answer{$\cos (2\theta)= (\cos{}(\theta))^{2}- (\sin{}(\theta))^{2} $}
\item $f(\theta)=\frac{\sec \theta}{1+\sec \theta}$.

\answer{$\frac{\sin{}(\theta)}{(1+\cos{}(\theta))^{2}} $}
\item $y=\frac{\cos x}{1-\sin x}$.

\answer{$\frac{1}{(1- \sin{}(x))} $}
\item $y=\frac{t\sin t}{1+t}$.

\answer{$ \frac{\sin{}(t)+t \cos{}(t)+t^{2}\cos{}(t) }{(1+t)^{2}}$}
\item $y=\frac{1-\sec x}{\tan x}$.

\answer{$\frac{\cos{}(x)- 1}{(\sin{}(x))^{2}} $}
\item $h(\theta)=\theta \csc \theta -\cot \theta$.

\answer{$\frac{1+\sin{}(t)- t \cos{}(t)}{(\sin{}(t))^{2}}$}
\item $y=x^2\sin x\tan x$.

\answer{$\frac{2 x \cos{}x \sin^2{}x+2 x^{2} \sin{}x  \cos^2x+x^{2} \sin^3{}(x)}{\cos^2{}x} $}
\end{enumerate}
\end{multicols}
