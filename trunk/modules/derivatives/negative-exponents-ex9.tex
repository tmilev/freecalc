\begin{frame}
\begin{example}[Example 9, p. 142]
\begin{columns}[c]
\column{.5\textwidth}
If $y = \frac{1}{x}$, then
\abovedisplayskip=0pt
\belowdisplayskip=0pt
\begin{eqnarray*}
\frac{\diff y}{\diff x} & = & %
\uncover<2->{\frac{\diff}{\diff x} (x^{-1})}\\
& \uncover<3->{ = } & %
\uncover<3->{-x^{-2}}\\
& \uncover<4->{ = } & %
\uncover<4->{-\frac{1}{x^2}}
\end{eqnarray*}
\column{.5\textwidth}
\abovedisplayskip=0pt
\belowdisplayskip=0pt
\begin{eqnarray*}
\frac{\diff}{\diff t}\left( \frac{6}{t^3}\right) & = & %
\uncover<5->{6\frac{\diff}{\diff t}(t^{-3})}\\
& \uncover<6->{ = } & %
\uncover<6->{6(-3t^{-4})}\\
& \uncover<7->{ = } & %
\uncover<7->{-\frac{18}{t^{4}}}\\
\end{eqnarray*}
\end{columns}
\end{example}
\end{frame}
