\begin{example}
Show that the graph of the given equation is a circle. Find the center and radius of the circle.

\begin{itemize}
\item $x^2+2x+y^2=1$.
\item $x^2+x+2y^2+y=1+y^2$.
\item $x^2=3x-y^2-2y$.
\item $3x^2+y=-3y^2$.
\item $2x^2+y=\frac{1}{2}x-2y^2$.
\end{itemize}
\end{example}
