\begin{definition}
Points on the graph of $ F(x,y)=G(x,y)$ for which 
\begin{itemize}
\item $x=0$ are called $y$-intercepts (those lie on the $y$ axis)
\item<2-> $y=0$ are called $x$-intercept (those lie on the $x$ axis). 
\end{itemize}
\end{definition}
