\begin{example}[Example 9, p. 401]
Find the area under the curve $y = e^{-3x}$ from 0 to 1.

\uncover<2->{
\[
A = \uncover<3->{\int_0^1 e^{-3x}\diff x} \uncover<4->{ = \left[ -\frac{1}{3}e^{-3x}\right]_0^1} \uncover<5->{ = \frac{1}{3}(1 - e^{-3}).}
\]
}
\end{example}
