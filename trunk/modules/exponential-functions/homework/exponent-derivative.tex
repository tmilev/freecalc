\pointsii{3}  $r(t) = Ae^{-kt^2}$, where $A$ and $k$ are unknown constants.  

\solution{%
\begin{align*}
r & = Ae^{-kt^2}. \\
\text{Let}\quad u & = -kt^2. \\
\text{Then}\quad r & = Ae^u. \\
\text{Chain Rule}\quad \frac{\diff r}{\diff t} & = \frac{\diff r}{\diff u}\frac{\diff u}{\diff t} \\
 & = (Ae^u)(-2kt) \\
 & = -2Akte^{-kt^2}.
\end{align*}
}%

\end{enumerate}
