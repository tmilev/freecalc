\solution{\ref{problemSimplify3^23^(-1)/(3^3sqrt(3^3))}.

\begin{align*}
\frac{3^2\cdot 3^{-1}}{3^3\cdot\sqrt{3^3}} & = \frac{3^2\cdot 3^{-1}}{3^3\cdot (3^3)^{\frac{1}{2}}} \\
 & = \frac{3^2\cdot 3^{-1}}{3^3\cdot 3^{\frac{3}{2}}} \\
 & = \frac{3^{2-1}}{3^{3+\frac{3}{2}}} \\
 & = \frac{3^{1}}{3^{\frac{9}{2}}} \\
 & = 3^{1-\frac{9}{2}} \\
 & = 3^{-\frac{7}{2}}.
\end{align*}
