Express each of the following as a single power.  

\begin{enumerate}
\item   $(2^5\cdot 2^7)/(2\sqrt{2})$

\pointsii{2} $(3^2\cdot 3^{-1})/(3^3\cdot \sqrt{3^3})$

\solution{%
\begin{align*}
\frac{3^2\cdot 3^{-1}}{3^3\cdot\sqrt{3^3}} & = \frac{3^2\cdot 3^{-1}}{3^3\cdot (3^3)^{1/2}} \\
 & = \frac{3^2\cdot 3^{-1}}{3^3\cdot 3^{3/2}} \\
 & = \frac{3^{2-1}}{3^{3+3/2}} \\
 & = \frac{3^{1}}{3^{9/2}} \\
 & = 3^{1-9/2} \\
 & = 3^{-7/2}.
\end{align*}
}%

\item   $\pi^3/(\pi^{-1}\sqrt{\pi^5})$

\end{enumerate}
