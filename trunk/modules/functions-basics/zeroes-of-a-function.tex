\begin{frame}

\begin{definition}
The zeroes of a function $f$ are the values of the argument $x$ for which $f(x)=0$.
\end{definition}

\begin{observation}
The zeroes of a function are the $x$-coordinates of the $x$ intercepts of the graph of the function.
\end{observation}

\end{frame}
