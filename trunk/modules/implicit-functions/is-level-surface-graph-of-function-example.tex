

\begin{frame}
\begin{itemize}
\item The equation $x^2+y^2+z^2 = 1$ implicitly defines $z=\sqrt{1-x^2 -y^2 }$ as the unique function $z=f(x,y)$ such that

\begin{itemize}
\item $x^2+ y^2+(f(x,y))^2 = 1$ for all $(x,y)$ in a disk around $(0,0)$;
\item $f(0,0) = 1$.
\end{itemize}
\item The equation $x^2+y^2+z^2 = 1$ implicitly defines $z=-\sqrt{1 -x^2- y^2 }$ as the unique function $z=f(x,y)$ such that
%
\begin{itemize}
\item $x^2+ y^2+(f(x,y))^2 = 1$ for all $(x,y)$ in a disk around $(0,0)$;
\item $f(0,0) = -1$.
\end{itemize}
\end{itemize}


\end{frame}
