\begin{frame}
\begin{definition}[(Partial) Factorization]
To (partially) factor a polynomial means to rewrite it as a product of two polynomials of smaller degree.
\end{definition}
\uncover<2->{
\begin{example}
\[
\begin{array}{rcl}
x^2-1&=&(x-1)(x+1)\\
x^3+8&=&(x+2)\left(x^2-2x+4\right)
\end{array}
\]
\end{example}
}
\end{frame}
