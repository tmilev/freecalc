\begin{frame}
\begin{definition}
Let $A$ be a set. The notation 
\[
x\alertNoH{2}{ \in} A
\] is read as

\begin{itemize}
\item $x$ \alertNoH{2}{belongs to} $A$ or
\item $x$ \alertNoH{2}{is an element of} $A$.
\end{itemize}
\end{definition}
\uncover<3->{
\begin{example}
\alertNoH{4,5}{Express the statement $\alertNoH{7}{-1<}\alertNoH{6}{x} \alertNoH{7}{\leq 2} $} using the $\in$ symbol and the interval notation.

\smallskip 

\hfil\hfil$
\fcAnswer{5}{\alertNoH{6}{x}\in \alertNoH{7}{(-1,2]}}
$

\smallskip 

\alertNoH{8,9}{Express the statement $\alertNoH{10}{x} \alertNoH{11}{<0} $ \alertNoH{13}{or} $ \alertNoH{12}{1\leq} \alertNoH{10}{x} \alertNoH{12}{<2}$} using the $\in$ symbol and the interval notation.

\smallskip 

\hfil\hfil$
\fcAnswer{9}{\alertNoH{10}{x}\in \alertNoH{11}{(-\infty,0)}  \alertNoH{13}{ \cup} \alertNoH{12}{[1,2)}}
$

\end{example}
}
\end{frame}
