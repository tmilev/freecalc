\begin{example}
Write the set of numbers $x$ satisfying $\alertNoH{2,3}{ 0\leq x\leq 2}$ in interval notation.

\hfil \hfil $
\fcAnswer{3}{\left[ 0, 2\right]}
$

\hfil 
\hfil 
\uncover<3->{
\begin{pspicture}(-2.1,-0.3)(3.1,0.6)
\psline(-2,0)(3,0)
\psline[linecolor=\fcColorGraph, linewidth=2pt](0,0)(2,0)
%\fcFullDot{-1}{0}
%\fcFullDot{1}{0}
\rput[b](0,0.3){$0$}
\rput[b](2,0.3){$2$}
\rput(0,0){$\color{\fcColorGraph} [$}
\rput(2,0){$\color{\fcColorGraph}]$}
\end{pspicture}
}
\end{example}
