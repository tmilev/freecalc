\begin{frame}

\begin{definition} Let $A$ and $B$ be sets.
\begin{itemize}
\item The union of $A$ and $B$ is the set consisting of the elements in $A$ and the elements in $B$, without additional elements.
\item The union of $A$ and $B$ is denoted by 
\[
\alertNoH{7}{\alertNoH{3,5}{A}\cup \alertNoH{4,6}{B}}
\]
\end{itemize}
\end{definition}
\uncover<2->{
\begin{example}

}
