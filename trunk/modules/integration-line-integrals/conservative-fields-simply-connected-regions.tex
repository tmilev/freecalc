\frametitle{Simply Connected Regions}
If $P_y(x,y) \neq Q_x(x,y)$, then $\fcv{F}$ is not a gradient field.

\noindent \fcQuestion{2}{If $P_y(x,y) = Q_x(x,y)$, is $\fcv{F}$ necessarily a gradient field?} \fcAnswer{3}{No:}
\abovedisplayskip 0pt
\belowdisplayskip 0pt
\[
\begin{array}{r@{~}c@{~}l}
\uncover<3->{\displaystyle \fcv{F} &=&\displaystyle  \alert<4,5>{-\frac{y}{x^2+y^2}} \fcv{i} + \alert<6>{ \frac{x}{x^2+y^2}} \fcv{j} = \alert<4,5>{P} \, \fcv{i} + \alert<6>{Q} \, \fcv{j} }\\
\displaystyle \fcQuestion{4}{ P_y} &\fcQuestion{4}{=}& \displaystyle \fcAnswer{5}{\alert<6>{ \frac{y^2-x^2}{(x^2 +y^2 )^2}}} \uncover<6->{\alert<6>{= Q_x}} \\
\uncover<7->{\displaystyle \oint_{C=\text{circ. around }(0,0)} \!\!\!\!\! \fcv{F} \cdot \fcv{\diff r} &=&\displaystyle  \oint_C \left(-\frac{y}{x^2+y^2}  \diff y +\frac{x}{x^2+y^2} \diff y\right) = 2\pi \neq 0 \; .}
\end{array}
\]

\uncover<8->{
\begin{definition}
A domain $D$ is called \emph{simply connected} if every closed loop in $D$ can be deformed (``lassoed'') to a point inside $D$.
\end{definition}
}
\uncover<18->{
\begin{theorem}
Let $\fcv{F}=P\, \fcv{i} + Q \fcv{j}$ and $P_y=Q_x$. Suppose $\fcv F$ is defined over a simply connected open set. Then $\fcv{F}$ is a gradient field.
\end{theorem}
}
