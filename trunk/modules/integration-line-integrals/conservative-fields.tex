\begin{definition}
A vector field $\fcv{F}$ is called \emph{conservative} if for any two points $A$ and $B$ and any two paths $C_1$ and $C_2$ from $A$ to $B$ we have
$\displaystyle \int_{C_1} \fcv{F} \cdot \fcv{\diff r} = \int_{C_2} \fcv{F}\cdot \fcv{\diff r}\; .$
\end{definition}
\end{columns}
\uncover<2->{
\begin{lemma}[alternative definition]
A vector field $\fcv{F}$ is conservative if and only if every point $A$ every path $C$ starting and ending at $A$ we have $\displaystyle\oint_{C} \fcv{F} \cdot \fcv{\diff r} = 0\; .$
\end{lemma}
}%
\uncover<3->{
\begin{proof}
The path $\alert<5>{ C = C_1 \cup \alert<4>{(-C_2)}}$ starts and ends at $A$ \uncover<6->{and therefore 
$\displaystyle \alert<8>{0=} \oint_{C} \fcv{F} \cdot \fcv{\diff r}\uncover<7->{ = \alert<8>{\int_{C_1} \fcv{F} \cdot \fcv{\diff r} - \int_{C_2} \fcv{F} \cdot \fcv{\diff r}} \; .}$ \uncover<8->{$\Leftrightarrow$ $\displaystyle  \alert<8>{\int_{C_1} \fcv{F} \cdot \fcv{\diff r} = \int_{C_2} \fcv{F} \cdot \fcv{\diff r}}$.}}
\end{proof}
