\begin{frame}
\frametitle{Exact 1-Forms}
Let $\fcv{F} = P\, \fcv{i} + Q\, \fcv{j}$ be a vector field and $\omega = \fcv{F} \cdot \fcv{\diff r} = P\diff x + Q \diff y$ be the corresponding 1-form. 
\abovedisplayskip 0pt
\belowdisplayskip 0pt
\[
\begin{array}{rclll}
\uncover<2->{\fcv{F} &=& \nabla f && \Rightarrow} \\
\uncover<3->{P&=&f_x}\\
\uncover<3->{Q&=&f_y &&\Rightarrow} \\
\uncover<4->{P\diff x + Q\diff y &=& f_x\diff x+f_y\diff y &&\Rightarrow }\\ \uncover<5->{\omega &=& \diff f}
\end{array}
\]
\uncover<6->{
\begin{definition} %
1-forms that are (total) differentials of functions are called \emph{exact}. 
\end{definition}
}
\uncover<7->{
\begin{center}
  $\fcv{F}$ is a gradient field $\Rightarrow$ the 1-form $\omega = \fcv{F} \cdot \fcv{\diff r}$ is exact
\end{center}
}
\uncover<8->{
\begin{theorem}[Net Change Theorem for Line Integrals]
If $C$ is a curve from $A$ and $B$, then
$\displaystyle \int_C \diff f = \int_C (\nabla f) \cdot \fcv{\diff r} = f(B) - f(A)$.
\end{theorem}
}
\end{frame}
