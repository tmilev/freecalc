Let $C$-smooth curve enclosing a region $D$. \uncover<2->{Let $\fcv T$ - unit tangent vector,} \uncover<3->{$\fcv{N}$- unit vector, normal to $C$ and pointing inwards towards $D$.} \uncover<4->{Let $\fcv F$ be smooth vector field.} \uncover<5->{Define outward flux of $\fcv F$ relative to $C$ via:}
\uncover<5->{
\[
\text{Outward Flux } = -\oint_C \fcv{F} \cdot \fcv{N} \diff s = -\oint_C \fcv{F} \cdot \fcv{\diff n} \; ,
\]
where $\fcv{\diff n} =\fcv{N} \diff s$ and $\diff s$ is the element of arclength.
}
\end{columns}

\uncover<6->{
\begin{definition}
The \emph{divergence} of a vector field $\fcv{F}$ at a point $P$ is the density of flux:
\[
\begin{array}{rcl}
\displaystyle (\divg \fcv{F})(P) &= &\displaystyle  \alert<7-11>{\lim\limits_{D \to \{P\}}} \frac{\text{Outward flux across boundary}}{\text{Area}(D)} \\ &= &\displaystyle  \alert<7-11>{ \lim \limits_{ D\to \{P\}} }  \frac{1}{ \text{Area} ( D)} \oint_C \fcv{F} \cdot \fcv{dn}
\end{array}
\]
\end{definition}
