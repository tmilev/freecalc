\begin{columns}

\column{0.6\textwidth}
Let $C$ be a circle of radius $R$ centered at the origin, oriented counterclockwise. Compute the integral 
\[
\oint_C - \frac{y}{x^2+y^2} \diff x+\frac{x}{x^2+y^2} \diff y
\]
\end{columns}
\uncover<2->{ In polar coord.:  
$
\left| \begin{array}{rcl}
\alert<4>{x}&\alert<4>{=}& \alert<4>{r\cos{\theta}}  \\
\alert<4>{y}&\alert<4>{=}& \alert<4>{r\sin{\theta}}
\end{array}\right.
\Rightarrow 
\left| \begin{array}{r@{}c@{}l}
\fcQuestion{2}{\alert<6>{\diff x}} &\fcQuestion{2}{\alert<6>{=}}& \fcAnswer{3}{ \alert<6>{ \cos{\theta} \diff r - r\sin{\theta} \diff \theta} }\\ 
\fcQuestion{2}{\alert<6>{\diff y}} &\fcQuestion{2}{\alert<6>{=}}& \fcAnswer{3}{ \alert<6>{ \sin{\theta} \diff r + r\cos{\theta} \diff \theta}}
\end{array}\right.\quad .
$
}

\only<1-11>{
\abovedisplayskip 0pt
\[
\begin{array}{rcl}
\uncover<4->{\displaystyle \alert<11>{-\frac{\alert<4>{y}}{\alert<5>{x^2+y^2}} \diff x + \frac{\alert<4>{x}}{\alert<5>{x^2+y^2}}\diff y }& \alert<0>{=}& \displaystyle  \alert<7,9>{-} \frac{\alert<4,7>{\alert<8>{r} \alert<9>{\sin\theta}} }{\alert<5,8>{r^2}} (\alert<6>{ \alert<7>{\cos{\theta} \diff r} \alert<9>{-} \alert<8>{r}\alert<9>{\sin{\theta} \diff \theta}})}\\
\uncover<4->{&&\displaystyle  + \frac{\alert<4,7>{\alert<8>{r} \alert<9>{ \cos \theta}}}{ \alert<5,8>{r^2}} ( \alert<6>{ \alert<7>{ \sin{\theta} \diff r} + \alert<8>{r} \alert<9>{\cos{\theta}  \diff \theta}})  }\\
\uncover<7->{&\alert<0>{=}& \alert<9>{(\alert<10>{\cos^2\theta+\sin^2\theta})\diff \theta}}\\
\uncover<10->{&\alert<11>{=}&\alert<11>{ \diff \theta }}
\end{array}
\]
}
