\frametitle{Parallelotope definition}
\begin{itemize}
\item Let $\fcv v_1= (v_{11},\dots, v_{1n}),\dots, \fcv v_{k}= (v_{k1}, \dots, v_{kn})$ be $k$ vectors in $n$-dimensional space, $k\leq n$.
\item<2-> Let $\mathcal R$ be the region \alert<2-9>{spanned by} the vectors, coefficients  $\in [0,1]$.
\item<3-> That is, $\mathcal R=\left\{ \alert<4-6>{ t_1 \fcv v_1} +\alert<7-9>{t_2\fcv v_2} +\dots+ t_k\fcv v_k | \alert<4-6>{t_1\in [0,1]}, \dots, t_k\in [0,1]\right\}$.
\uncover<10->{
\begin{definition}[parallelotope]
We call a region $\mathcal R$ of the above form a $k$-dimensional parallelotope in $n$-dimensional space.
\end{definition}
}
\end{itemize}
