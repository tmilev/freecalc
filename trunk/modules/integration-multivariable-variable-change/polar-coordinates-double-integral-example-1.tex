\uncover<6->{ $\fcv f:\left|\begin{array}{rcl}\alert<11,12>{x}&\alert<11,12>{=}&\alert<11,12>{r\cos \theta }\\\alert<11,12>{y}&\alert<11,12>{=}&\alert<11,12>{r \sin \theta } \end{array} \right. .$ Let $\mathcal R$ = parametr. of $\mathcal S$ in polar coord., $\fcv f(\mathcal R)=\mathcal S$.} 

\uncover<7->{
\hfil
$
\fcQuestion{7}{\mathcal{R}=} \left\{(r,\theta) | \fcAnswer{8}{\alert<14>{ 1 \leq r \leq 2, \frac{\pi}{2} \leq \theta \leq \frac{3\pi}{2}}} \right\} \uncover<9->{= [1,2]\times \left[\frac{\pi}{2}, \frac{3\pi}{2}\right]}\; .
$
\hfil

$
\begin{array}{rcl}
\uncover<10->{\iint_{\mathcal S}\alert<11,12>{(x+y)} \alert<10>{ \diff x \diff y} &=&\iint_{\alert<10>{\mathcal R}} \uncover<10,11>{ \alert<11>{ \textbf{?}}} \uncover<12->{\alert<12>{r(\sin \theta+\cos\theta )}} \alert<10>{r~ \diff r\diff \theta}} \\
