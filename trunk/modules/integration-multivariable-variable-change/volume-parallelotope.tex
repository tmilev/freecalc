\begin{theorem}[$k$-volume = \alert<5>{Gram determinant}]
$\Vol_k(\mathcal R_k)=\sqrt{\alert<5>{
\left| \begin{array}{ccc}
\fcv v_1\cdot \fcv v_1 & \dots&\fcv v_1\cdot \fcv  v_{k}\\
\vdots & \dots &\vdots\\
\fcv v_{k}\cdot \fcv v_{1}& \dots &\fcv v_k \cdot \fcv v_k 
\end{array}  \right|}}$.
\end{theorem}
}\uncover<6->{Proof: studied in Linear algebra ($\Vol_k$ - defined by algebra only). }\uncover<7->{$\Vol_n(\mathcal R_n)$ is a perfect square for all $n$.}
