\column{0.4\textwidth}
Let $\mathcal{R}$ be the region bounded by $\alertNoH{8,17}{y=x-1}$ and $\alertNoH{8,17}{y^2=2x+6}$. Compute
\[
\iint_{\mathcal{R}}\left(2+\frac{1}{4}xy \right) \diff x \diff y.
\]
\end{columns}
\only<handout:1|1-18>{%
\fcQuestion{2}{Plot $x-1$.} \fcQuestion{4}{Plot $y^2=2x+6$.} \uncover<6->{\alertNoH{6}{Identify the region.}} \fcQuestion{7}{The two curves intersect when} \fcAnswer{8}{$
\begin{array}{rcl}
(x-1)^2&=&2x+6 \\
x^2-2x+1&=&2x+6\\
x^2-4x-5&=&0\\
x&=& -1 \text{ or } 5 .
\end{array}
$}

\uncover<9->{\alertNoH{9,10,11,12}{The two intersection points are $(-1, \fcAnswer{10}{\alertNoH{15}{-2}})$ and $(5, \fcAnswer{12}{\alertNoH{15}{4}})$.}} \uncover<13->{The function can be plotted as above.} \uncover<14->{The integral becomes:}
}%only<1-18>

\uncover<14->{%
$
\begin{array}{r@{~}c@{~}l}
\displaystyle
\alertNoH{18,19}{\int_{{\fcQuestion{14}{y=} \fcAnswer{15}{-2}} }^{{ \fcQuestion{14}{y=} \fcAnswer{15}{4} }} {\alertNoH{20,21}{\int}}_{{ \fcQuestion{16}{x=} \fcAnswer{17}{ \frac{y^2-6}{2}}} }^{{ \fcQuestion{16}{x=} \fcAnswer{17}{y+1} }} \alertNoH{20,21}{ \left(2 + \frac{1}{4}xy \right)} \alertNoH{20,21}{\diff x} \diff y} \uncover<handout:2| 20->{ &=&\displaystyle \int_{y=-2}^{y=4} {\left[ \fcAnswer{21}{\alertNoH{22,23}{2x+ \frac{x^2y}{8}}} \right]}_{\alertNoH{22,23}{x=\frac{y^2-6}{2}} }^{\alertNoH{22,23}{x=y+1}} \diff y }\\
\uncover<handout:2|22->{&=&\displaystyle {\alertNoH{24,25}{\int}}_{y =- 2 }^{ y=4} \fcAnswer{23}{ \alertNoH{24,25}{\left(-\frac{1}{32} y^{5}+\frac{1}{2} y^{3} \right. }} }\\
\uncover<handout:2|22->{&& \displaystyle \phantom{ {\int}_{ y=-2}^{y=4}}~~ \uncover<23->{ \alertNoH{23,24,25}{ \left.-\frac{3}{4} y^{2} + y+8  \right) }} \alertNoH{24,25}{\diff y} }\\
\uncover<handout:2|24->{&=& \left[\fcAnswer{25}{ - \frac{1}{192} y^{6}+\frac{1}{8} y^{4}}\right.} \\
\uncover<24->{&&  { ~\left.\uncover<25->{\alertNoH{25}{ -\frac{1}{4} y^{3}+\frac{1}{2} y^{2}+8 y }} \right] }_{-2}^{4}\uncover<26->{=45}}
\end{array}
$
}%
