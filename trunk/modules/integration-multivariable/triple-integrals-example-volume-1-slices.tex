\vbox to 4.2cm{

Compute the volume of the \alertNoH{3-7}{region $\cR$ bounded by} \alertNoH{7,12,13,14,15 }{$ x + 2y+z=2$}, \alertNoH{6,14,15}{$x=2y$}, \alertNoH{5,12,13}{$x=0$}, \alertNoH{4}{$z=0$}.

\uncover<2->{
\hfil $\displaystyle \text{vol}(\cR) = \iiint_{\cR} 1\cdot \diff V .$
}

\uncover<3->{\alertNoH{3-7}{$\cR$ is}} \uncover<3-7>{\alertNoH{3-7}{\textbf{?}}} \uncover<8->{\alertNoH{8}{a tetrahedron with vertices at $(0,0,0)$, $(0,1,0)$, $(0,0,2)$, and $\left(1, \frac{1}{2}, 0\right)$}.}

}
\end{columns}
\only<1-17>{
\uncover<9->{Project $\cR$ onto the $z-$axis to get segment from $z=0$ to $z=2$.} \fcQuestion{10}{Fix a value for $z$ to get the slice $S_z$ } \fcAnswer{11}{shown in the picture.}
}
\uncover<16->{
\alertNoH{17,18}{
$\displaystyle \text{vol}(\cR) = \iiint_{\cR} 1\cdot \diff V = \int_{z=0}^{z=2} \left( \iint_{S_z} 1\cdot \diff x \diff y \right)  \diff z
$
}
}

\only<19->{
Project $S_z$ onto $x$-axis to get segment from $x=0$ to $x=1-\frac{z}{2}$.} \uncover<21->{ Fix $x\in[0, 1-\frac{ z }{ 2}]$.} \fcQuestion{22}{Vertical slice: segment from $y= \fcAnswer{ 23}{ \frac{ x }{2 }}$ to $y=\fcAnswer{23}{ 1- \frac{ z}{2} - \frac{x}{2}}$.}
\uncover<24->{
$\displaystyle
\text{vol}(\cR) = \iiint_{\cR} 1\cdot \diff V = \int_{ \alertNoH{25}{ z=0} }^{ \alertNoH{25}{z=2} } \left( \int_ { \alertNoH{26}{ x=0}}^{\alertNoH{26}{ x=1-\frac{z}{2} }} \left( \int_{ \alertNoH{27}{ y= \frac{ x}{2} }}^{\alertNoH{27}{ y=1 -\frac{ z}{2} -\frac{x}{2}}} 1 \cdot \diff y \right)  \diff x\right) \diff z .
$
}
