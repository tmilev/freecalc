\begin{frame}
\frametitle{Iterated Integrals}
\begin{itemize}
\item To compute a triple integral over $\cR$ one reduces to iterated integrals.
\item<2-> One reduces to
\begin{itemize}
\item<2-> a \alertNoH{4}{single integral of a double integral}
\item<3-> or \alertNoH{8}{double integral of a single integral}.
\end{itemize}
\item<4-> Single integral of a double integral: \alertNoH{4}{decomposition into slices}.
\uncover<5->{
\begin{itemize}
\item \alertNoH{6}{Project the body on an axis.}
\item \alertNoH{7}{Look at 2D slices perpendicular to that axis (CT-scan).}
\end{itemize}
\[
\iiint_{\cR}f(P)  \diff V = \alertNoH{6}{\int_{\text{location of slice}} } \left(\alertNoH{7}{\iint_{\text{slice}} f(P) \diff A} \right) \alertNoH{6}{\diff h}
\]
}
\item<8-> Double integral of a single integral: decomposition into rods.
\begin{itemize}
\item\alertNoH{9}{Project the body on a plane}.
\item\alertNoH{10}{Look at 1D slices perpendicular to that plane (rods)}.
\end{itemize}
\[
\iiint_{\cR}f(P) \diff V = \alertNoH{9}{\iint_{\text{location of rod}}}  \left( \alertNoH{10}{\int_{\text{rod}} f(P)  \diff h} \right)  \diff A
\]
\end{itemize}
\end{frame}
