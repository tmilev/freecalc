\begin{theorem}[The Fundamental Theorem of Calculus]
Let $f$ be a function continuous on $[a, b]$ and let $G(x) = \displaystyle \int_a^x f(t) \diff t$ for all $x\in [a,b]$. Then $G$ is differentiable and $G'(x) = f(x)$.  
\end{theorem}

Let $\varepsilon $ be an (arbitrarily small) positive number. Since $f$-continuous, there exists $\delta$ such that $|f(t)-f(x)|<\varepsilon$ for all $t$ for which $|x-t|<\delta$. In other words, $f(x)+\varepsilon> f(t)>f(x)- \varepsilon$ for all $t$ for which $|x-t|<\delta$. Then 
\[
h(f(x)+\varepsilon)=(x+h-x)(f(x)+\varepsilon)=\int\limits_{x}^{x+h} f(x)+\varepsilon dt >\int_{x}^{x+h}f(t)dt>\int\limits_{x}^{x+h} f(x)+\varepsilon dt =  (x+h-x)(f(x)-\varepsilon)=h(f(x)-\varepsilon)
\]

\uncover<1-9>{
\[
\begin{array}{l}
G'(x)=\lim\limits_{h\to 0} \frac{G(x+h)-G(x)}h\uncover<1->{=\lim\limits_{h\to 0} \frac{\alert<4>{\int \limits_{a}^{ x+h} f(t)dt}-\alert<5>{\int\limits_{a}^{x}f(t)dt}}h} \\
\phantom{G'(x)}=\lim\limits_{h\to 0} \frac{\int\limits_x^{x+h}f(t)dt}{h}
\end{array}
\]
