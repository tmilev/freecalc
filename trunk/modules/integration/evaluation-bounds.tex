% begin module indefinite-integral
\begin{frame}
We often use the notation
\[
\left. F(x)\right]_a^b = F(b)-F(a)
\]
or 
\[
\left[ F(x)\right]_a^b = F(b)-F(a)
\]
Therefore we can write
\[
\int_a^b f(x) \diff x = \left. F(x)\right]_a^b
\]
or
\[
\int_a^b f(x) \diff x = \left[ F(x)\right]_a^b
\]
\end{frame}
% end module indefinite-integral
