Integrate.
\begin{enumerate}
\item $\displaystyle \int \sin ^5 x\diff x$

\answer{$\displaystyle -\frac{\cos^5 x}{5}+\frac{2\cos^3 x}{3}-\cos x +C$}

\item $\displaystyle \int \arcsin x \diff x$

\answer{$\displaystyle x\Arcsin x +\sqrt{1-x^2}+C$}

\item $ \displaystyle \int x^3\sqrt{1-x^2}\diff x$

\answer{$\displaystyle -\frac{\left(\sqrt{1 -x^2}\right)^3}{3} +\frac{\left( \sqrt{1 -x^2}\right)^5 }{5} +C$} 

\item $\displaystyle x\cos (3x)\diff x$

\answer{$\displaystyle \frac{x\sin (3x)}{3}+\frac{\cos (3x)}{9} +C$}

\item $\displaystyle \tan^2 t \sec^4 t\diff t$

\answer{ $ \displaystyle \frac{\tan^3 t}{3}+\frac{\tan^5 t}{5}+C$}

\item $\displaystyle \int \frac{\diff t}{t^2\sqrt{t^2-16}}$

\answer{$ \displaystyle \frac{\sqrt{t^2-16}}{16t} +C$}
\end{enumerate}
