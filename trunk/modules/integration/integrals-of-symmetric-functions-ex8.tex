\begin{example}[Example 8, p. 338]
Since $f(x) = x^6 + 1$ satisfies $f(-x) = f(x)$, it is even, and so
\begin{eqnarray*}
\int_{-2}^2 (x^6+1) \ \diff x %
& = &%
\uncover<2->{%
2\int_0^2 (x^6+1) \ \diff x%
}\\%
& \uncover<3->{ = } &%
\uncover<3->{%
2\left[ \frac{1}{7}x^7 + x\right]_0^2%
}\\%
& \uncover<4->{ = } &%
\uncover<4->{%
2\left( \frac{128}{7} + 2\right)%
}\\%
& \uncover<5->{ = } &%
\uncover<5->{%
\frac{284}{7}%
}%
\end{eqnarray*}
\end{example}
\end{frame}
