\begin{problem} (Textbook, page 410, problems 59-64)
Find the inverse function. 
\begin{multicols}{2}
\begin{enumerate}
\item $y=ln (x+3)$.

\answer{$f^{-1}(x)=e^x-3$}
\item $y=2^{10^x}$.

\answer{$f^{-1}(x)=\log_2(\log_{10} x), \quad x>1$}
\item $f(x)=e^{x^3}$.

\answer{$f^{-1}(x)=\sqrt[3]{\ln x}, \quad x>0$}
\item $y=(\ln x)^2$, $x\geq 1$.

\answer{$f^{-1}(x)=e^{\sqrt{x}}, \quad x\geq 0 $}
\item $y=\log_{10}\left(1+\frac{1}{x} \right)$, $x\in (-\infty, -1)\cup (0, \infty)$.

\answer{$f^{-1}(x)=\frac{1}{10^x-1}\quad , \quad x\neq 0$}
\item $y=\frac{e^x}{1+2e^x}$.

\answer{$f^{-1}(x)= \ln \left(\frac{x}{1-2x}\right) $, \quad $x\in (0, \frac12) $}
\end{enumerate}
\end{multicols}
\end{problem}
\begin{problem}
Find the inverse function. You are asked to do the algebra only; you are not asked to determine the domain and range of the function or its inverse. 
\begin{multicols}{2}
\begin{enumerate}
\item $f(x)= 3x^2+4x-7$, where $x\geq -\frac{2}{3}$.

\answer{$f^{-1}(x)= -\frac{2}3+\frac{\sqrt{25+3x}}{3}, \quad x\geq -\frac{25}{3}$}
\item $f(x)= 2x^2+3x-5$, where $x\geq -\frac{3}{4}$.

\answer{$f^{-1}(x)=-\frac{3}{4}+\frac{\sqrt{49+8x}}{4}, \quad x\geq -\frac{49}{8}$}
\item $f(x)=ax^2+bx+c$, where $x\leq -\frac{b}{2a}$.

\answer{$f^{-1}(x)=\frac{-b}{2a}-\frac{\sqrt{b^2-4ac+4ax}}{2a}, \quad x\geq \frac{-b^2+4ac}{4a} $}
\item $f(x)= \frac{2x+5}{x-4}$, $x\neq 4$.

\answer{$f^{-1}(x)=\frac{4x+5}{x-2}, \quad x\neq 2$}
\item $f(x)= \frac{3x+5}{2x-4}$, $x\neq 2$.

\answer{$f^{-1}(x)=\frac{4x+5}{2x-3}, \quad x\neq \frac32$}
\item $f(x)=\frac{ax+b}{cx+d}$, $x\neq -\frac{d}c$, $ad-bc\neq 0$.

\answer{$f^{-1}(x) =\frac{d x - b }{ -x c+a}, \quad x\neq \frac{a}{c}$}
\item $f(x)=\frac{e^x- e^{-x}}{e^x+e^{-x}}$.

\answer{$f^{-1}(x) =\frac{1}{2}\ln \frac{1+x}{1-x},\quad  x\in (-1, 1)$ }
\item $f(x)=2^{2x}+2^{x}-2$.

\answer{$f^{-1}(x) =\log_2\frac{-1+\sqrt{9+4x}}{2}, \quad x\geq -2$}
\item $f(x)=3^{4x}- 5* 3^{2x}+4$, $x\leq \log_{3} \sqrt{ \left( \frac{5}{2}\right)}$.

\answer{$f^{-1}(x) =\frac{1}{2}\log_3 \frac{5-\sqrt{9+4x }}{2}, \quad x> 4 $}
\end{enumerate}
\end{multicols}
\end{problem}
