\begin{example}
\begin{columns}
\column{0.4\textwidth}
Find $f^{-1}(x)$ for $f(x) = \frac{e^x-e^{-x}}{e^x+e^{-x}}  $. 
\uncover<16->{\alertNoH{16}{$f=\tanh$ = hyperbolic tangent function.}}

\psset{xunit=0.7cm, yunit=0.7cm}
\begin{pspicture}(-3.05,-3.1)(3.15,3.2)
\psframe*[linecolor=white](-3.05,-3.1)(3.15,3.2)
\tiny
\psaxes[ticks=none, labels=none]{<->}(0,0)(-3.05,-3.1)(3.05,3.1)
\fcLabels{3.05}{3.1}
%Function formula: (679570457/250000000^{x}- (679570457/250000000^{- (x)}))/(679570457/250000000^{- (x)}+679570457/250000000^{x})
\uncover<1-14>{
\rput(-2,-0.5){$y=f(x)$}
\psplot[linecolor=\fcColorGraph, plotpoints=1000]{-3}{3}{2.71828 x -1 mul exp -1 mul 2.71828 x exp add 2.71828 x exp 2.71828 x -1 mul exp add div }
}
\uncover<15->{
\rput(-2,-0.5){$y=f(x)$}
\psplot[linecolor=gray!50, plotpoints=1000]{-3}{3}{2.71828 x -1 mul exp -1 mul 2.71828 x exp add 2.71828 x exp 2.71828 x -1 mul exp add div }
}
%Function formula: 1/2 (\ln{}((1+x)/(1- (x))))
\uncover<15->{
\rput[tl](1.1,2.5){$y=f^{-1}(x)$}
\psplot[linecolor=\fcColorGraph, plotpoints= 1000]{ -0.994}{ 0.994 }{x 1 add x -1 mul 1 add div ln 0.5 mul }
}
\end{pspicture}
\uncover<15->{Final }\uncover<14->{answer}\uncover<15->{, \alertNoH{15}{relabelled}:}
\[
\uncover<14->{
f^{-1}( \only<handout:0|14>{y}\only<15->{\alertNoH{15}{x}} )=\frac12\ln\left( \frac{1+\only<handout:0|14>{y} \only<15->{\alertNoH{15}{x}} }{1-\only<handout:0|14>{y} \only<15->{\alertNoH{15}{x}}}\right)
}
\]
\column{0.6\textwidth}
\uncover<2->{Set $\alertNoH{13}{u=e^x}$.} \uncover<3->{Then $\alertNoH{3}{e^{-x}=} \fcAnswer{4}{\frac{1}{e^x}= \frac{1}{u}}$.}
$ \begin{array}{r@{~}c@{~}l}
\displaystyle\uncover<5->{ \frac{u-\frac1u}{u+\frac1u}&=&y}\\
\displaystyle\uncover<6->{\frac{u^2-1}{u^2+1}&=&y}\\
\displaystyle\uncover<7->{u^2-1&=&y(u^2+1)}\\
\displaystyle\uncover<8->{u^2(1-y)&=&1+y}\\
\displaystyle\uncover<9->{u^2&=&\displaystyle \frac{1+y}{1-y}}\\
\displaystyle \uncover<10->{u&=&\displaystyle \only<10>{\pm} \only<handout:0|11>{ \alertNoH{11}{+}} \only<handout:0|12->{  \phantom{ +~\,} } \sqrt{\frac{1+y}{1-y}}}\\
\displaystyle \uncover<12->{\uncover<13->{\alertNoH{13}{ x=}} \alertNoH{13}{\alertNoH{12}{\ln} u} &=&\displaystyle \alertNoH{12}{\ln} \sqrt{\frac{1+y}{1-y}}} \uncover<14->{=\frac12\ln\left(\frac{1+y}{1-y}\right)} \\
\end{array}
$

\uncover<11->{where the sign follows from $e^x=u>0$.} \uncover<12->{Take $\alertNoH{12}{\ln} $ on both sides.}
\end{columns}
\end{example}
