\column{.5\textwidth}
\begin{enumerate}
\item \alertNoH{ 2-3}{Domain: \fcAnswer{3}{\worksheet{$[-1, 1]$}.}}
\item \alertNoH{ 4-5}{Range: \fcAnswer{5}{\worksheet{$[0, \pi ]$}.}}
\item  $\Arccos x = y \Leftrightarrow \cos y = x$ and $0 \leq y \leq \pi$.
\item  $\Arccos (\cos x) = x$ for $0 \leq x \leq \pi$.
\item  $\cos (\Arccos x) = x$ for $-1 \leq x \leq 1$.
\item  $\displaystyle \frac{\diff}{\diff x} (\Arccos x) = -\frac{1}{\sqrt{1-x^2}}$.  \uncover<6->{(The proof is similar to the proof of the formula for the derivative of $\Arcsin x$.)}
\end{enumerate}
