\begin{example}
Find $\Arcsin (\sin 1.5)$.  
\begin{itemize}
\item<2->  \alert<2-3>{$\frac{\pi}{2} \approx \uncover<3->{1.57.}$}
\item<4->  Therefore $-\frac{\pi}{2} \leq 1.5\leq \frac{\pi}{2}$.  
\item<5->  Therefore $\Arcsin (\sin 1.5) = 1.5$.  
\end{itemize}
\end{example}
