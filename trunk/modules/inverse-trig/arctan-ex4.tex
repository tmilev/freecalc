\begin{frame}
\begin{example} %[Example 4, p. 219]
Evaluate 
\[
\lim_{x\rightarrow 2^+} \arctan \left( \frac{1}{x-2}\right) .
\]
\uncover<2->{
\[
\frac{1}{x-2} \rightarrow \infty \qquad \text{ as } \qquad x\rightarrow 2^+.
\]
}
\uncover<3->{
Therefore 
\[
\lim_{x\rightarrow 2^+} \arctan \left( \frac{1}{x-2}\right) = \frac{\pi}{2}.
\]
}
\end{example}
\end{frame}
