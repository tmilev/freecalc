Compute the derivative (derive the formula).

\begin{multicols}{2}
\begin{enumerate}
\item $(\Arctan x) '$. \answer{$\frac{1}{1+x^2}$}
\item $(\Arccot x)'$. \answer{$-\frac{1}{1+x^2}$}
\item $(\Arcsin x)'$. \answer{$\frac{1}{\sqrt{1-x^2}}$}
\item $(\Arccos x)'$. \answer{$-\frac{1}{\sqrt{1-x^2}}$}
\item Let $\text{arcsec}$ denote the inverse of the secant function. Compute $(\text{arcsec} x)'$.
\answer{$\frac{1}{x\sqrt{x^2-1}}$}
\end{enumerate}
\end{multicols}
