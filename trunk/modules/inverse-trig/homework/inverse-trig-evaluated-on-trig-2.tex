Find each of the following values.  Express your answers precisely, not as decimals.  
\begin{enumerate}
\item $\arcsin(\sin 4)$.

\answer{$\pi-4 $}
\item $\arcsin(\sin 0.5)$.

\answer{$ 0.5$}
\item $\arcsin(\cos 120^\circ)$.

\answer{$ -\frac{\pi}{6}$}
\item $\arccos(\cos (3))$.

\answer{$3 $}
\item $\arccos(\cos (-2))$.

\answer{$ 2$}
\item $\arccos(\sin (-4))$.

\answer{$\frac{3\pi}{2}-4\approx 0.712389$}
\item \label{problemarctan(tan5)}  $\arctan(\tan 5)$. 

\answer{$ 5-2\pi$}
\end{enumerate}
