Evaluate the limit, or show that it does not exist.

\begin{multicols}{2}
\begin{enumerate}
\item $\lim\limits_{x\to 0}\frac{x^2}{1-\cos x} $
\answer{$2$}
\item $\lim\limits_{x\to\infty}x \tan \left(\frac{1}{x}\right) $
\answer{$1$}
\item $\lim \limits_{x\to 0^+}x^{\sqrt{x}}$
\answer{$1$}
\end{enumerate}
\end{multicols}
