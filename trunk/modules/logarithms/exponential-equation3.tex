\begin{example}[Finding the intersection of two exponential graphs]
Find the point(s) of intersection of $y = e^{2x}+4$ and $y = 5e^x$.  
\abovedisplayskip=0pt
\belowdisplayskip=0pt
\begin{align*}
\uncover<2->{ e^{2x} + 4 & = 5e^x }\\
\uncover<3->{\alert<5-6>{e^{2x}} - 5\alert<4>{e^x} + 4} & \uncover<3->{ = 0} \\
\intertext{\uncover<4->{Substitute $\alert<10>{u = e^x}$:}}
\uncover<4->{\alert<5-6>{\uncover<6->{u^2}} - 5\alert<4>{u} + 4} & \uncover<4->{ = 0} \\
\uncover<7->{\alert<7-8>{(\uncover<8->{u-4})(\uncover<8->{u-1})}} & \uncover<7->{ = 0 } 
\end{align*}
\begin{align*}
\uncover<9->{ \alert<10>{u} } & \uncover<9->{ = 4 } & \uncover<9->{\text{or}} & & \uncover<9->{\alert<10>{u}} & \uncover<9->{ = 1} \\
\uncover<10->{ \alert<10>{e^x} } & \uncover<10->{ = 4 } & \uncover<10->{\text{or}} & & \uncover<10->{\alert<10>{e^x}} & \uncover<10->{ = 1} \invisible{99999999} \\
\uncover<11->{ \alert<11-12>{x} } & \uncover<11->{ \alert<11-12>{ = \uncover<12->{\ln 4} }} & \uncover<11->{\text{or}} & & \uncover<11->{\alert<13-14>{x}} & \uncover<11->{ \alert<13-14>{ = \uncover<14->{0} }} 
\end{align*}
\uncover<15-19>{Therefore the points of intersection are \alert<15-16>{$(\ln 4,\uncover<16->{20})$} and \alert<17-18>{$(0,\uncover<18->{5})$}.}  
\end{example}
