\solution{\ref{problemd/dx((3x-2)/(-2x+3))}

\[
\begin{array}{rcll|l}
\displaystyle \frac{\diff}{\diff x} \left(\ln{}\left(\frac{3 {{x}}-2}{-2 {{x}}+3}\right)\right) &=& \displaystyle \frac{\diff }{\diff x} \left(\ln (3x-2)-\ln(-2x+3) \right)&&\text{logarithm properties}\\
&=&\displaystyle \frac{(3x-2)'}{(3x-2)}-\frac{(-2x+3)'}{-2x+3}&& \text{chain rule}\\
&=&\displaystyle \frac{3}{3x-2}-\frac{-2}{-2x+3}\\
&=&\displaystyle \frac{3}{3x-2}-\frac{2}{2x-3}\\
&=&\displaystyle \frac{-5}{6 x^{2}-13 x+6}&&\text{combine fractions (optional).}
\end{array}
\]
