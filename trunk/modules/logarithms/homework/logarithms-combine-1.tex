\answer{$-\ln 2=\ln \left(\frac{1}{2}\right) $}

\answer{$3$}

\item $\log_7(24)+\log_{\frac{1}{7}}3-\log_{49} (64)$.

\item $\log_3(24)+\log_{3}\left(\frac{3}{8}\right)$.

\end{enumerate}
