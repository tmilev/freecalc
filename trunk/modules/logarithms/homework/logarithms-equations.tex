
\begin{problem}(Textbook, page 409, problem 27-36)
Solve each equation for $x$. Using a calculator give an ($\approx$) answer in decimal notation. Using calculator verify your approximate solutions.
\begin{multicols}{2}
\begin{enumerate}
\item $e^{6-4x}=6$.

\answer{$\frac{6-\ln 6 }{4}$}
\item $\ln (3x-10)=2$.
\item $\ln (x^2-1)=3$.
\item $e^{2x}-3e^x+2=0$.
\item $2^{x-5}=3$.
\item $\ln x+\ln (x-1)=1$.
\item $e^{3x+1}=k$.
\item $\log_2(mx)=c$.
\item $e- e^{-2x}=1$.
\item $10(1+e^{-x})^{-1}=3$.
\item $\ln (\ln x)=1$.
\item $e^{e^x}=10$.
\item $e^{2x}-e^x-6=0$.
\item $\ln(2x+1)=2-\ln x$.
\end{enumerate}
\end{multicols}
\end{problem}
