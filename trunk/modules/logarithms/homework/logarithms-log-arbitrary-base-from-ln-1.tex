Using only the $\ln$ operation of your calculator compute the indicated logarithm. Confirm your computation numerically by exponentiation.


\begin{multicols}{2}
\begin{enumerate}[ref={\fcProblemRef}]
\item  $\log_{5}(13)$. 

\answer{$\frac{\ln 13}{\ln 5} \approx 1.593693$}
\item  $\log_{12}(9)$. 

\answer{$\frac{\ln 9}{\ln 12} \approx 0.884228$}
\item  $\log_{13}(101)$. 

\answer{$\frac{\ln 101}{\ln 13} \approx 1.799303$}
\item  $\log_{10}(2015)$. 

\answer{$\frac{\ln 2015}{\ln 10} \approx 3.304275$}

\end{enumerate}
\end{multicols}
