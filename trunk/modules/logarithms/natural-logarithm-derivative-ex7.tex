\begin{example}[Example 7, p. 413]
Find $f'(x)$ if $f(x) = \ln |x|$.
\uncover<2->{
\[
\textrm{Since }\qquad f(x) = \left\{ \begin{array}{lr}
\ln x & \textrm{ if } x > 0 \\
\ln (-x) & \textrm{ if } x < 0 \\
\end{array}\right.
\]
}
\uncover<3->{
\[
\textrm{it follows that }\qquad f'(x) = \left\{ \begin{array}{lr}
\frac{1}{x} & \textrm{ if } x > 0 \\
\frac{1}{-x}(-1) = \frac{1}{x} & \textrm{ if } x < 0 \\
\end{array}\right.
\]
}
\uncover<4->{
Therefore $f'(x) = \frac{1}{x}$ for all $x\neq 0$.
}
\end{example}
\end{frame}
