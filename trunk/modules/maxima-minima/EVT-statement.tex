\begin{frame}[t]
\begin{theorem}[The Extreme Value Theorem]
If $f$ is continuous on a closed interval $[a,b]$, then $f$ attains its absolute maximum value $f(c)$ and its absolute minimum value $f(d)$ at some numbers $c$ and $d$ in $[a,b]$.
\end{theorem}
