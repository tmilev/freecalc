Find the maximal volume of a box with no lid whose surface area is $10m^2$.

\uncover<2->{ Let the three dimensions of the box be $\alert<2>{ x,y,z}$.} 
\end{columns}
\uncover<3->{We seek to maximize $V=xy \alert<6>{z}$.}
\uncover<4->{The restrictions are: $\alert<5>{xy+2(zx+zy)=10} $.} \uncover<5->{Therefore $ \alert<5,6>{z= \frac{ 10 - x y}{2(x+y)}}$. }\uncover<6->{Therefore $V=xy\alert<6>{\frac{10-xy }{2(x+y)} } $.} \uncover<7->{We need to solve the system:

$
\begin{array}{rcl}
\alert<12>{0=} \alert<8,9>{ V_x}&\uncover<8->{\alert<8,9>{=}}& \uncover<8>{\alert<8>{\textbf{?}}} \uncover<9->{\alert<9>{ \frac{ \alert<12>{ -x y^{3}-\frac{1}2x^{2}y^{2}+5y^{2}}}{(x+y)^2}}} \\
\alert<13>{0=} \alert<10,11>{V_y}&\uncover<10->{\alert<10,11>{=}} & \uncover<10>{ \alert<10>{\textbf{?}}}\uncover<11->{\alert<11>{ \frac{ \alert<13>{ -y x^{3} - \frac{1}2x^{2}y^{2}+5x^{2}} }{(x+y)^2}}}
\end{array}
$
}

\uncover<12->{
We can assume $y\neq 0$, $x\neq 0$ (else the volume is zero). Then 
$
\begin{array}{rcl}
\alert<12>{0}&\alert<12>{=}&\alert<12>{-x y-\frac{1}2x^{2}+5} \\
\alert<13>{0}&\alert<13>{=}&\alert<13>{-y x-\frac{1}2y^{2}+5}\\
\end{array}
$
}

\uncover<14->{
Therefore $x^2=y^2$ and so $x=y$ (both quantities are positive).} \uncover<15->{ Therefore $\frac{3}{2}x^2=5$,} \uncover<16->{and so $x=\sqrt{\frac{10}{3}}=y$.} \uncover<17->{By EVT max exists} \uncover<18->{$\Rightarrow$ is achieved for $x=y=\sqrt{\frac{10}{3}}$.} \uncover<19->{Max volume $V_{max}=  \frac{5}{9}\sqrt{30} $.}
