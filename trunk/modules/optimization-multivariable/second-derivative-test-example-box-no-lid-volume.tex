Find the maximal volume of a box with no lid whose surface area is $10m^2$.

\uncover<2->{ Let dimensions of box be $\alert<2>{ x,y,z}$.} \uncover<3->{We seek to maximize $V=xy \alert<6>{z}$.}
\end{columns}
\uncover<4->{Restrictions: $\alert<5>{xy+2(zx+zy)=10} $.} \uncover<5->{$\Rightarrow $ $ \alert<5,6>{z= \frac{ 10 - x y}{2(x+y)}}$. }\uncover<6->{$\Rightarrow$ $V=xy\alert<6>{\frac{10-xy }{2(x+y)} } $.} 
\fcQuestion{7}{$(x,y)\in$ } \fcAnswer{8}{$\mathcal R=\{(x,y)|xy\leq 10, x\geq 0, y\geq 0 \}$.}
\uncover<9->{We're solving:

$
\begin{array}{rcl}
\alert<14>{0=} \alert<10,11>{ V_x}&\uncover<10->{\alert<10,11>{=}}& \fcAnswer{11}{ \frac{ \alert<14>{ -x y^{3}- \frac{1}{2} x^{2} y^{2} + 5 y^{2}}}{(x+y)^2}} \\
\alert<15>{0=} \alert<12,13>{V_y}&\uncover<12->{\alert<12,13>{=}} & \fcAnswer{13}{\frac{ \alert<15>{ -y x^{3} - \frac{1}2x^{2}y^{2}+5x^{2}} }{(x+y)^2}}
\end{array}
$
}

\uncover<14->{
We can assume $y\neq 0$, $x\neq 0$ (else the volume is zero). Then 
$
\begin{array}{rcl}
\alert<14>{0}&\alert<12>{=}&\alert<14>{-x y-\frac{1}2x^{2}+5} \\
\alert<15>{0}&\alert<13>{=}&\alert<15>{-y x-\frac{1}2y^{2}+5}\\
\end{array}
$
}

\uncover<16->{
Therefore $x^2=y^2$ and so $x=y$ (both quantities are positive).} \uncover<17->{ Therefore $\frac{3}{2}x^2=5$,} \uncover<18->{and so $x=\sqrt{\frac{10}{3}}=y$.} \uncover<19->{By EVT max exists} \uncover<20->{$\Rightarrow$ is achieved for $x=y=\sqrt{\frac{10}{3}}$.} \uncover<21->{Max volume $V_{max}=  \frac{5}{9}\sqrt{30} $.}
