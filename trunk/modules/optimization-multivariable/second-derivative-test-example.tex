\column{0.7\textwidth}
Find the local and global maxima and minima of $f(x,y) = x^4+y^4-4xy$.
\[
\begin{array}{rcl}
\alert<2,3>{f_{xx}}&\alert<2,3>{=}& \uncover<3->{\alert<3>{12x^2}}\\
\alert<4,5>{f_{xy}}&\alert<4,5>{=}& \uncover<5->{\alert<5>{-4}}\\
\alert<6,7>{f_{yy}}&\alert<6,7>{=}& \uncover<7->{\alert<7>{12y^2}}\\
\alert<8,9>{D= f_{xx}f_{yy}-f_{xy}^2}&\alert<8,9>{=}& \uncover<9->{\alert<9>{ 144 x^2 y^2 - 16}.}
\end{array}
\]
\uncover<10->{ The critical points were previously computed.}
\end{columns}
\uncover<10->{ 
\begin{tabular}{|c|c|c|c|c|c|}
\hline
$(x_0,y_0)$ & \alert<11>{$f_{xx}$} & \alert<11>{$f_{yy}$} & \alert<11>{$f_{xy}$} & \alert<11>{$D$} & \alert<13,14,17,18,21,22>{Extremum ?} \\\hline
\alert<10-14>{$(0,0)$} & \uncover<11>{\alert<11>{\textbf{?}}} \fcUncoverAlert{12}{$0$} &\fcAnswer{12}{$0$} & \fcAnswer{12}{$-4$} & \fcAnswer{12}{$-16 <0$} & \fcAnswer{14}{Saddle point}  \\\hline
\alert<10,15-18>{$(1,1)$} & \fcAnswer{16}{$12$} & \fcAnswer{16}{$12$} & \fcAnswer{16}{$-4$} & \fcAnswer{16}{$128 > 0$} & \fcAnswer{18}{Local min}  \\\hline
\alert<10,19-22>{$(-1,-1)$} & \fcAnswer{20}{$12$} & \fcAnswer{20}{$12$} & \fcAnswer{20}{$-4$} & \fcAnswer{20}{$128 > 0$} & \fcAnswer{22}{Local min} \\\hline
\end{tabular}
}

\uncover<handout:1,2|26->{ In this case it turns out that the two local minimum points are actually global minimum points, because
$f(x,y) = x^4+y^4-4xy = (x^2-1)^2+(y^2-1)^2+2(x-y)^2 -2  \geq -2\; .$
}
