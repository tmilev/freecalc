\uncover<2->{
Set $h$ - cone height,} \uncover<3->{$t$ - cone radius.} \uncover<4->{Then $\alert<4>{V=} \uncover<5->{\alert<5>{\frac{1}3 (\alert<6>{\text{area cone base}})h} }\uncover<6->{= \frac13 \alert<6>{\pi   t^2} h }$.} \uncover<7->{ Let $\alert<7>{\theta}$ - angle of the wedge.} \uncover<8->{Then $ \alert<8,9>{\text{arc}{AB}=} \uncover<9->{\alert<9>{r\theta}}$ \uncover<10->{\alert<10,11>{= perimeter cone base =}} $\uncover<11->{\alert<11>{2\pi t}}$.} Therefore $t=\frac{r\theta}{2\pi}$. Then \[h= \sqrt{r^2-t^2 } = \sqrt{r^2-\frac{r^2\theta^2}{4\pi^2}}=\frac{r}{2\pi }\sqrt{ 4\pi^2-\theta^2 },\] and therefore 
\[
\begin{array}{rcl}
V&=&\displaystyle\frac13\pi ht^2= \pi \frac{r^2\theta^2}{4\pi^2}\frac{r}{2\pi}\sqrt{4\pi^2-\theta^2}\\ ~\\
&=& \displaystyle \frac{r^3}{8\pi^2} \theta^2\sqrt{4\pi^2-\theta^2}\quad . 
\end{array}
\]
