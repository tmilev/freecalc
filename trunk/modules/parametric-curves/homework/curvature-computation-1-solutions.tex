\begin{array}{rcll|l}
\fcv r(t)&=&\displaystyle  (t\cos t, t\sin t, -t)\\
\fcv r'(t)&=&\displaystyle (\cos t -t\sin t, \sin t+t\cos t, -1)\\
\fcv r''(t)&=&\displaystyle \left(- t \cos{}t-2 \sin{}t, - t \sin{}t+2 \cos{}t, 0\right)\\

&=&\displaystyle \sqrt{t^{4}+5 t^{2}+8}\\
