Find $\frac{\diff y}{\diff x}$ and $\frac{\diff^2 y}{\diff x^2}$ for the following parametric curves.
\begin{enumerate}
\item $\displaystyle \left|\begin{array}{rcl} x&=&t^3+1\\y&=&t^2-t\end{array}\right. $

\psset{xunit=0.5cm, yunit=0.5cm}
\begin{pspicture}(-2.775, -0.9)(9.316377,4.25) 
\tiny 
\psaxesStandard{-2.525}{-0.65}{9.066377}{3.9}

%Calculator input: plotCurve{}(t^{3}+1, t^{2}- t, -3/2, 2)
\parametricplot[linecolor=\psColorGraph, plotpoints=1000]{-1.5}{2}{ 1 t 3 exp add t -1 mul t 2 exp add }
\end{pspicture}

\answer{$\displaystyle \frac{\diff y}{\diff x}= \frac{2}{3}t-\frac{1}{3}t^2$, $\displaystyle \frac{\diff^2 y}{\diff x^2}=-\frac{2}{9}t^4+\frac{2}{9} t^5 $}
\item $\displaystyle \left|\begin{array}{rcl} x&=&t^2+1\\y&=&e^t-1\end{array}\right. $

\psset{xunit=0.5cm, yunit=0.5cm}
\begin{pspicture}(-0.9, -1.317915)(7.65,6.822919) 
\tiny 
\psaxesStandard{-0.65}{-1.067915}{7.4}{6.472919}

%Calculator input: plotCurve{}(t^{2}+1, e^{t}-1, -5/2, 2)
\parametricplot[linecolor=\psColorGraph, plotpoints=1000]{-2.5}{2}{ 1 t 2 exp add -1 2.718281828 t exp add }
\end{pspicture}

\answer{
$\displaystyle \frac{\diff y}{\diff x}= \frac{e^{t}}{2t} $, $\displaystyle \frac{\diff^2 y}{\diff x^2}=\frac{e^t(t-1)}{4t^3} $
}
 
\end{enumerate}
