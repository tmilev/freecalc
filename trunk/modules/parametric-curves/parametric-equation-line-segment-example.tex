\begin{frame}
\begin{example}
Parametrize
\begin{itemize}
\item the line $L$ passing through $P_0(1,2,3)$ and $P_1(5,2,1)$;
\item the line segment $S$ connecting $P_0(1,2,3)$ and $P_1(5,2,1)$.
\end{itemize}
\uncover<2->{Direction of $L$:} \uncover<3->{$\fcv{u} = \fcv{r}_1-\fcv{r}_0 = ( 4, 0, -2)$.}\uncover<4->{Parametric vectorial equations of $L$:}
\uncover<5->{%
\[
\fcv{r} = ( 1,2,3) + t ( 4, 0 -2) \Leftrightarrow \fcv{r} = ( 1+4t, 2, 3-2t)\; .
\]
}
\uncover<6->{
Parametric scalar equations of line $L$:
\[
\left\{ \begin{array}{ll}
x & = 1+4t \\
y & = 2 \\
z & = 3-2t
\end{array}
\right. , \quad t \text{ real number.}\]
} %
\uncover<7->{
Parametric vectorial equation of segment $S$:
\[
\fcv r= t(1,2,3) +(1-t)(5,2,1)\quad t\in [0,1]\quad .
\]
}
\end{example}
\end{frame}
