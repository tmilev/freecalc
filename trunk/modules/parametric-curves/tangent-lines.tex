$\begin{array}{rcl}
\displaystyle \fcv{f}\colon [a,b]&\to&\displaystyle  \RR^3\\ 
\displaystyle \fcv{f}(t) &=& \displaystyle ( x(t), y(t), z(t))\\
\displaystyle \fcv{f}'(t)&=&\displaystyle \alertNoH{2-6}{ \lim_{h\to 0}\frac{\fcv f(t+h)- \fcv f(t)}{h} } \\
&=&\displaystyle ( x'(t), y'(t), z'(t)) \\
\end{array}
$

\vskip -0.1cm
\begin{itemize}
\item $\fcv{f}\,'(t_0)$: direction of tangent line through $\fcv f(t_0)$.
\item<7-> Tangent equation at $\fcv{f}(t_0)$:

\hfil\hfil$
\fcv{r}(t) = \fcv{f}(t_0) + t \fcv{f}\,'(t_0) .
$
\item<8-> Linear approximation:

\hfil\hfil $
\fcv{f}(t) \approx L_{\fcv{f},t_0} (t) = \fcv{f}(t_0) + t \fcv{f}\,'(t_0).
$
\item<9-> A linear approximation is good if:

\hfil\hfil $\displaystyle
\lim_{t\to 0} \left|\frac{\fcv{f}(t) - L_{\fcv{f},t_0}(t)}{t} \right| =0.
$
\item<10-> Differentials:

\hfil\hfil$
\diff \fcv{f} = \fcv{f}' \diff t = ( x',y',z' )\diff  t  .
$
\end{itemize}
