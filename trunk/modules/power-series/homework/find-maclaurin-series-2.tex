Find the Maclaurin series of the function. The answer key has not been proofread, use with caution.
\begin{enumerate}
\item $\frac{1}{2x+3}$.

\answer{ $ \frac{1}{3}\left(1- \frac{2x}{3} +\left(\frac{2 x}{ 3} \right)^2 -\left(\frac{2x}{3}\right)^3+\dots\right) = \sum\limits_{0}^{ \infty} \frac{(-1)^n}{3}\left(\frac{2}{3}\right)^n x^n  $}
\item $\frac{1}{(1-x)^2}$.

\answer{ $1+2x+3x^2+4x^3+\dots=\sum\limits_{n=0}^{\infty} nx^n$}
\item $\frac{1}{(1-x)^3}$.

\answer{ $\frac{1}{2}\left( 2+6x+12x+\dots+n(n-1)x^{n-2}+\dots \right) =\sum\limits_{n=0}^{\infty} \frac{(n+1)(n+2)}{2} x^n$}
\item $x e^{-2x}$.

\answer{ $\sum\limits_{n=0}^{\infty} (-1)^n 2^{n} x^{n+1}= \sum\limits_{n=1}^{\infty} (-1)^{n-1} 2^{n-1} x^{n}$}
\end{enumerate}
