\answer{ $\begin{array}{l}
\displaystyle \ln 3-\sum\limits_{n=1}^\infty \left( \frac{ 2}{3}\right)^n \frac{x^{2n }}{n} \\
\displaystyle R=\sqrt{\frac{2}{3}}\\
\displaystyle \text{converges for }x\in \left(-\sqrt{\frac{2}{3}}, \sqrt{\frac{2}{3}} \right)
\end{array}
$}
