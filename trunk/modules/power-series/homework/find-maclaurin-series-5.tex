Compute the Maclaurin series of the function.
\begin{multicols}{2}
\begin{enumerate}
\item $\frac{1}{1+x^2}$.
\item $\frac{1}{1-2x^2}$.
\item $\frac{1}{x^2-1}$.
\item $\frac{\frac12}{x-1}-\frac{\frac12}{x+1}$.
\item $ \frac{1}{(1-x)^2}$.
\item $ \frac{1}{(1-x)^3}$.
\item $\ln (1+x)$.
\item $\ln (1-x)$.
\item $\ln (1-3x)$.
\item $\ln (1-3x^2)$.
\item $\arctan x$.
\item $\arctan (2x)$.
\item $\arctan \left(2x^2\right)$.

\end{enumerate}
\end{multicols}
