Compute the Maclaurin series of the function.
\begin{multicols}{2}
\begin{enumerate}[ref={\fcProblemRef}]
\item \label{problemMaclaurinsqrt(1+x)} $\displaystyle \sqrt{1+x}$.

\answer{$ \sum\limits_{n=0}^\infty \binom{\frac{1}{2}}{n} x^n$}
\item \label{problemMaclaurin(1+x)^(-1/2)} $\displaystyle \frac{1}{\sqrt{1+x}}$.

\answer{$ \sum\limits_{n=0}^\infty \binom{-\frac{1}{2}}{n} x^n$}
\item \label{problemMaclaurin(1-x^2)^(-1/2)} $\displaystyle \frac{1}{ \sqrt{ 1- x^2 }}$.

\answer{$ \sum\limits_{n=0}^\infty (-1)^n\binom{- \frac{1}{2}}{n} x^{2n}$}
\item \label{problemMaclaurin(arcsin x)} $\displaystyle \arcsin x$.

\answer{$ \sum\limits_{n=0}^\infty (-1)^n\binom{- \frac{1}{2}}{n} \frac{x^{2n+1 } } { 2n + 1}$}
\end{enumerate}
\end{multicols}
