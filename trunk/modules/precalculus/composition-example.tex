& & f\circ g \\
& & \uncover<2->{(f\circ g)(x)}\\
& \uncover<3->{ = } & \uncover<3->{f(\alert<handout:0| 4>{g(x)})}\\

&& \uncover<7->{
\textrm{Domain}:}\\
&& \uncover<7->{
(-\infty , 2].
}
\end{eqnarray*}

[0 , 4].
}
\end{eqnarray*}
