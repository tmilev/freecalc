\begin{problem}Graph the functions by hand, by applying consecutively the transformations learned in class.
\begin{multicols}{2}
\begin{enumerate}
\item $y=\frac{1}{x}$.
\item $y=\frac{1}{x+1}$.
\item $y=\frac{1}{2x+1}$.
\item $y=\frac{3}{2x+1}$.
\item $y=\frac{3+x}{2x+1}$.
\item $y=\left|\frac{3+x}{2x+1}\right|$.
\end{enumerate}
\end{multicols}
\end{problem}
