Solve the quadratic equation. The answer key has not been proofread, use with caution.
\begin{enumerate}[ref={\fcProblemRef}]
\item $2x^2-5x+3=0$.

\answer{$x_1=1$, $x_2=\frac{3}{2}$}
\item $2x^2+6x+3=0$.

\answer{$x_1,x_2=\frac{-3\pm \sqrt{3}}{2}$}
\item $3x^2-5x-2=0$.

\answer{$x_1=-\frac{1}{3}$, $x_2=2$}
\item $3x^2-5x-3=0$.

\answer{$x_1,x_2=\frac{5\pm \sqrt{61}}{6}$}
\end{enumerate}
