
\begin{frame}
\frametitle{$Ax^2+By^2+Cz^2+D=0$, $A>0, B>0, C>0, D>0$}
Consider the surface 
$\mathcal C =\{(x,y,z) |Ax^2+By^2+Cz^2+D=0 \}$.

\begin{itemize}
\item Let $A>0, B>0, C>0, D>0$.
\item<2-> Then the surface is the empty set.
\item<3-> Example: 
\[
x^2+2y^2+3z^2+4=0.
\]
\end{itemize}
\vskip 10cm

\end{frame}
