\column{0.7\textwidth}

\alertNoH{4}{10 ft ladder} rests against a vertical wall.  \alertNoH{ 6}{The bottom of the ladder slides away from the wall at a rate of 1 ft/s}. \alertNoH{ 8}{How fast is the ladder top sliding down when the bottom is 6 ft from the wall?}
\end{columns}

\begin{columns}[c]
\column{.55\textwidth}
\begin{itemize}
\item<2->  Let \alertNoH{2}{$y$= dist. from top to ground}.
\item<2->  Let \alertNoH{3}{$x$= dist. from bottom to wall}.
\item<5-| alert@5-6,16>  Given: $\fcAnswerUncover{4}{6}{ \frac{\diff x}{\diff t} = 1}$ \uncover<6->{ft/s.}
\item<5-| alert@6-7> Unknown: \fcAnswerUncover{4}{8}{$\frac{\diff y}{\diff t}$ \alertNoH{ 18}{when $x = 6$ ft}.}
\item<20-| alert@20>  Pythagorean Therem: $y =\sqrt{10^2-6^2}= 8$.
\item<9-| alert@10-11>  Relationship b/n quantities.
\item<9-| alert@12>  Differentiate (use Chain Rule).
\end{itemize}

\column{.45\textwidth}
\abovedisplayskip=0pt
\belowdisplayskip=0pt
\abovedisplayshortskip=0pt
\belowdisplayshortskip=0pt
\begin{align*}
\uncover<11->{%
\alertNoH{11}{x^2+y^2}%
}%
& \uncover<11->{\alertNoH{ 11}{ = }} %
\uncover<11->{%
\alertNoH{11}{10^2=100}%
}\\%
\uncover<12->{%
2x\frac{\diff x}{\diff t} + 2y\frac{\diff y}{\diff t}%
}%
& \uncover<12->{ = } %
\uncover<12->{%
0%
}\\%
\uncover<13->{%
\frac{\diff y}{\diff t}%
}%
& \uncover<13->{ = } %
\uncover<13->{%
-\frac{\alertNoH{ 17-18}{x}}{\alertNoH{19-20}{y}}\alertNoH{ 15-16}{\frac{\diff x}{\diff t}}%
}\\%
\uncover<14->{%
\frac{\diff y}{\diff t}%
}%
& \uncover<14->{ = } %
\uncover<14->{%
-\frac{\fcAnswerUncover{14}{18}{ 6\text{ ft} }}{\fcAnswerUncover{14}{20}{8\text{ ft}}}\cdot\alertNoH{ 15-16}{\fcAnswerUncover{14}{16}{ 1\text{ ft/s}}}%
}\\%
& \uncover<21->{ = } %
\uncover<21->{%
\alertNoH{22}{-}3/4 \text{ ft/s.}%
}%
\end{align*}
\uncover<22->{%
Therefore the top of the ladder is \alertNoH{22}{falling} at a rate of $3/4$ ft/s.
}%
