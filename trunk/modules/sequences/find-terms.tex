\begin{example}[Sequences via formulas: find sequence terms]

\begin{enumerate}

\[

}%
\]
\item<alert@4-5| handout:alert@0>  $b_n = 1$
\[
\uncover<5-| handout:0>{%
1,1,1,1,1,\ldots
}%
\]
\item<alert@6-7| handout:alert@0>  $c_n = -3(n-1)+5$
\[
\uncover<7-| handout:0>{%
5,2,-1,-4,-7,\ldots
}%
\]
\item<alert@8-9| handout:alert@0>  $d_n = n^2+1$
\[
\uncover<9-| handout:0>{%
2,5,10,17,26,\ldots
}%
\]
\end{enumerate}
\end{example}
