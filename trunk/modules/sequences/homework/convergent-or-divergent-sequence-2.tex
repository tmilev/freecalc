\begin{enumerate}
Find the limit of the sequence or prove that the sequence is divergent.
\begin{itemize}
\item $a_n=\left(\frac{n}{n-1}\right)^{2n}$.
\answer{convergent, $\lim\limits_{n\to \infty} a_n = e^2$}
\item $a_n=\frac{n!}{n^n}$.
\answer{convergent, $\lim\limits_{n\to \infty} a_n=0$}
\end{itemize}
\end{enumerate}
