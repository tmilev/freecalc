Determine if the sequence is convergent or divergent. If convergent, find the limit of the sequence.
\begin{multicols}{2}
\begin{enumerate}
\item $a_n=n$.
\answer{divergent}
\item $a_n=2^n$.
\answer{divergent}
\item $a_n=1.0001^n$.
\answer{divergent}
\item $a_n=0.999999^n$.
\answer{convergent, $\lim_{n\to \infty} a_n=0$}
\item $a_n=n-\sqrt{n+1}\sqrt{n+2}$
\answer{convergent, $\lim_{n\to \infty} a_n=-\frac{3}{2}$}
\item $a_n=\frac{\ln n}{n}$.
\answer{convergent, $\lim_{n\to \infty} a_n=0$}
\item $a_n=\frac{\ln n}{\sqrt[10]{n}}$.
\answer{convergent, $\lim_{n\to \infty} a_n=0$}
\item $a_n=\frac{1}{n}$.
\answer{convergent, $\lim_{n\to \infty} a_n=0$}
\item $a_n=\frac{1}{n!}$.
\answer{convergent, $\lim_{n\to \infty} a_n=0$}
\item $a_n=\frac{n^n}{n!}$.
\answer{divergent}
\item $a_n=\cos n$.
\answer{divergent}
\item $a_n=\cos\left(\frac{1}{n}\right)$
\answer{convergent, $\lim_{n\to \infty} a_n=1$}

\item $a_n= \left(\frac{n+1}{n}\right)^{n}$.
\answer{convergent, $\lim_{n\to \infty} a_n=e$}
\item $a_n= \left(\frac{2n+1}{n}\right)^{n}$.
\answer{divergent}

\item $a_n= \left(\frac{n+1}{n}\right)^{2n}$.
\answer{convergent, $\lim_{n\to \infty} a_n=e^2$}

\item $a_n= \left(\frac{n+1}{2n}\right)^{n}$.
\answer{convergent, $\lim_{n\to \infty} a_n=0$}
\end{enumerate}
\end{multicols}
