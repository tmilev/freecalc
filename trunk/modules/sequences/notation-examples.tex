% begin module notation-examples
\begin{frame}
\begin{example}
The sequence 
\[
\left( -1, 1, -1, 1, -1, 1, \ldots \right)
\]
can be written $b_n = (-1)^n$.  
\end{example}

\uncover<2->{
\begin{example}
The sequence 
\[
\left( 1,2,4,8,16,\ldots\right)
\]
can be written $c_n = 2^{n-1}$.  
\end{example}
}

\uncover<3->{
\begin{example}
The sequence 
\[
\left( \frac{1}{2}, -\frac{1}{4},\frac{1}{8},-\frac{1}{16},\ldots\right)
\]
can be written $d_n = -\big( -\frac{1}{2}\big)^n$.  
\end{example}
}
\end{frame}
% end module notation-examples
