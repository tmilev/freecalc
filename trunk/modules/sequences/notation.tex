\begin{frame}
\begin{example}[Sequence notation]
Consider the sequence
\[
2, 4, 6, 8, \ldots.
\]
We can express this sequence more compactly using the notation
\[
a_n = 2n,
\]
where $a_n$ denotes the $n$th term.  
\begin{align*}
\text{So} \quad a_1 & = 2\cdot 1 = 2 \\
a_2 & = 2\cdot 2 = 4 \\
a_3 & = 2\cdot 3 = 6 \\
a_4 & = 2\cdot 4 = 8 \\
 & \vdots
\end{align*}
\end{example}

\end{frame}
