\begin{theorem}[Convergence of Geometric Sequences]
The sequence $\{ r^n\}$ is convergent if $-1 < r \leq 1$ and divergent otherwise.
\[
\lim_{n\to\infty}r^n = \left\{ \begin{array}{lll}
0 & \textrm{ if } & -1 < r < 1\\
1 & \textrm{ if } & r = 1\\
\end{array}\right.
\]
\end{theorem}
