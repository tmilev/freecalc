\begin{example}
The sequence $\left\{ \frac{1}{2n+1}\right\}$ is decreasing because
\[
a_n = \frac{1}{2n+1} \qquad a_{n+1} = \frac{1}{2(n+1)+1} = \frac{1}{2n+3}
\]
and 
\[
\frac{1}{2n+1} > \frac{1}{2n+3}
\]
because the denominator of the latter is bigger.
\end{example}
\end{frame}
