\begin{frame}
\frametitle{Alternating Series}
\begin{definition}[Alternating Series]
An alternating series is a series whose terms are alternately positive and negative.
\end{definition}
\uncover<2->{%
\begin{examples}
Here are two examples:
\abovedisplayskip=0pt
\[
\begin{array}{r@{\ }c@{\ }c@{\ }c@{\ }c@{\ }c@{\ }c@{\ }c@{\ }c@{\ }c@{\ }c@{\ }c@{\ }c@{\ }l}
\alertNoH{ 3-6}{1} &%
\alertNoH{ 7-8}{-} &%
\frac{\alertNoH{ 5-6}{1}}{\alertNoH{ 3-4}{2}} &%
 + &%
\frac{\alertNoH{ 5-6}{1}}{\alertNoH{ 3-4}{3}} &%
\alertNoH{ 7-8}{-} &%
\frac{\alertNoH{ 5-6}{1}}{\alertNoH{ 3-4}{4}} &%
 + &%
\frac{\alertNoH{ 5-6}{1}}{\alertNoH{ 3-4}{5}} &%
\alertNoH{ 7-8}{-} &%
\frac{\alertNoH{ 5-6}{1}}{\alertNoH{ 3-4}{6}} &%
 + &%
\cdots &%
\displaystyle  = \sum_{n=1}^\infty \uncover<8->{\alertNoH{ 8}{(-1)^{n-1}}}\frac{\uncover<6->{\alertNoH{ 6}{1}}}{\uncover<4->{\alertNoH{ 4}{n}}} \\%
\alertNoH{ 13-14}{-} %
\frac{\alertNoH{ 11-12}{1}}{\alertNoH{ 9-10}{2}} &%
 + &%
\frac{\alertNoH{ 11-12}{2}}{\alertNoH{ 9-10}{3}} &%
\alertNoH{ 13-14}{-} &%
\frac{\alertNoH{ 11-12}{3}}{\alertNoH{ 9-10}{4}} &%
 + &%
\frac{\alertNoH{ 11-12}{4}}{\alertNoH{ 9-10}{5}} &%
\alertNoH{ 13-14}{-} &%
\frac{\alertNoH{ 11-12}{5}}{\alertNoH{ 9-10}{6}} &%
 + &%
\frac{\alertNoH{ 11-12}{6}}{\alertNoH{ 9-10}{7}} &%
\alertNoH{ 13-14}{-} &%
\cdots &%
\displaystyle  = \sum_{n=1}^\infty \uncover<14->{\alertNoH{ 14}{(-1)^{n}}}\frac{\uncover<12->{\alertNoH{ 12}{n}}}{\uncover<10->{\alertNoH{ 10}{n+1}}} %
\end{array}
\]
\end{examples}
}%
