Express the sum of the series as a rational number.
\begin{multicols}{2}
\begin{enumerate}
\item 
\label{problemSum(2^n+3^n)/(5^n)}
\[
\sum\limits_{n=1}^{\infty} \frac{2^n+3^n}{5^n}
\]

\answer{$\frac{13}{6}$}

\item \label{problemSum(3^n+5^n)/(7^n)}
\[
\sum\limits_{n=1}^{\infty} \frac{5^n-3^n}{7^n}
\]

\answer{$\frac{7}{4}$}
\end{enumerate}
\end{multicols}
