\begin{example}[The sum of an infinite geometric series]
Find the sum of the geometric series
\abovedisplayskip=0pt
\belowdisplayskip=0pt
\[
\sum_{n=1}^{\infty} \frac{1}{2^n}.
\]
\uncover<2->{%
\begin{align*}
\alertNoH{ 2-3}{\text{The ratio is } r} & \alertNoH{ 2-3}{= \uncover<3-| handout:0>{\frac{1}{2}.}} \\
\alertNoH{ 4-5}{ |r|} & \alertNoH{ 4-5}{\uncover<5-| handout:0>{<}1.} \\
\alertNoH{ 6-7}{\text{The first term is } a} & \alertNoH{ 6-7}{= \uncover<7-| handout:0>{\frac{1}{2}.}} \\
\alertNoH{ 8-9}{\text{The sum is } s} & \alertNoH{ 8-9}{= \uncover<9-| handout:0>{\frac{1}{2}\cdot\frac{1}{1-\frac{1}{2}} = 1.}}
\end{align*}
}%
\end{example}

\end{frame}
