\begin{frame}
\frametitle{ Series}
\begin{definition}[Series]
If we add the terms in an infinite sequence, we get an infinite series:
\[
a_1 + a_2 + a_3 + a_4 + \cdots + a_n + \cdots 
\]
We denote this sum by
\[
\sum_{n = 1}^\infty a_n \qquad \textrm{ or } \qquad \sum a_n
\]
\end{definition}
