Evaluate the definite integral 
\begin{enumerate}
\item $\displaystyle\int\limits_{1}^{2} \frac{x}{2x^2+1 }  \diff x$.

\answer{$\frac14 \ln 3$}
\item $\displaystyle\int\limits_{0}^{\frac{1}4}\frac{x }{\sqrt{1-3x^2}}\diff x$.

\answer{$\frac{1}3\left(1-\sqrt{\frac{13}{16}} \right)$}
\end{enumerate}
