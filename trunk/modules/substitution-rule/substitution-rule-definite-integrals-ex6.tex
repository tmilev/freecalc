\begin{itemize}
\item<2-> Let \alertNoH{ 3-4,10,12,14}{$u = \fcAnswerUncover{2}{4}{2x+1}.$}
\item<2-> Then \alertNoH{ 5-6}{$\diff u = \fcAnswerUncover{2}{6}{2 \diff x}$.}
\item<7-> Therefore \alertNoH{ 7-8,15}{$\diff x = \fcAnswer{8}{\frac{1}{2}\diff u}$.}
\item<9-> When \alertNoH{ 9-10,16}{$x = 0$, $u = \fcAnswer{10}{1}$.}
\item<9-> When \alertNoH{ 11-12,17}{$x = 4$, $u =\fcAnswerUncover{9}{12}{9}$.}
\end{itemize}
