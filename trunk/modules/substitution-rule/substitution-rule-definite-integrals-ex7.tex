\begin{example}
Find $\displaystyle \int_{\alertNoH{9,10}{1} }^{\alertNoH{11,12}{2}} \frac{ \alertNoH{7, 8}{\diff x}}{(\alertNoH{ 4}{2-3x})^2}$.
\begin{itemize}
\item<2-> Let \alertNoH{3-4,10,12,14}{$u =\fcAnswerUncover{2}{4}{2-3x}$.}
\item<2-> Then \alertNoH{5-6,8}{$\diff u = \fcAnswerUncover{2}{6}{-3 \ \diff x}$.}
\item<7-> Therefore \alertNoH{7-8,15}{$\diff x =\fcAnswer{8}{-\frac{1}{3}\diff u}$.}
\item<9-> When \alertNoH{9-10, 16}{$x = 1$, $u = \fcAnswer{10}{-1}$.}
\item<9-> When \alertNoH{11-12, 17}{$x = 2$, $u = \fcAnswerUncover{9}{12}{-4}$.}
\end{itemize}
