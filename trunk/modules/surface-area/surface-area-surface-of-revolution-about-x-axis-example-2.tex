\column{0.65\textwidth}
Find the surface area of the ellipsoid obtained by rotating $y=\sqrt{1-\frac{x^{2}}{2}}$ about the $x$ axis.

\uncover<handout:2|14>{
\[
\begin{array}{rcl}
y'&=& \frac{1}{2}\frac{\left(-x\right)}{\sqrt{1-\frac{x^2}{2}}} = -\frac{x}{2y}\\
(y')^2&=&\frac{x^2}{4y^2}\\

\end{array}
\]
}
\end{columns}
\uncover<handout:2|14>{
\[
\begin{array}{rcl}
Area&=& \int_{-2}^22\pi y \sqrt{1+\frac{x^2}{4 y^2}}\diff x\\
&=&\int_{-2}^{2}2\pi y \sqrt{\frac{4y^2+x^2}{4y^2}}\diff x\\
&=&\int_{-2}^{2}2\pi y \sqrt{\frac{4y^2+x^2}{4y^2}}\diff x\\
&=&\int_{-2}^2\pi\sqrt{4\left(1-\frac{x^2}{2} \right)+x^2 }\diff x\\
&=&\int_{-2}^{2}\pi\sqrt{4-x^2}\diff x \\
\end{array}
\]
}
