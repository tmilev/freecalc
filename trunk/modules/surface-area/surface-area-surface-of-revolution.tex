\begin{itemize}
\item Let $C$ be parametrized curve in the $x>0$ half plane of the $xz-$plane, parametrized by $x=h(u)$, $z=g(u)$, $a \leq u \leq b$.
\item<2-> Let $S$ be the surface obtained by revolving $C$ about the $z-$axis.
\end{itemize}

\end{columns}
\uncover<6->{
Parametrize $S$:
$\left|\begin{array}{rcl}
x&=&h(v)\cos{u}\\ 
y&=&h(v)\sin{u}\\
z&=&g(v)
\end{array}\right.,u \in[0,2\pi], v\in[a,b] $

\[
\begin{array}{rcl}
\displaystyle \diff S &=& \displaystyle |\fcv f_u \times \fcv f_v| \diff u \diff v = h(v) \sqrt{(h'(v)^2+(g'(v))^2}  \diff u \diff v\\
\displaystyle \text{Area}(S) &= & \displaystyle  \iint\limits_{[a,b]\times [0,2\pi]} h(v) \sqrt{(h'(v))^2+(g'(v))^2}  \diff u \diff v  \\
&= &\displaystyle  2\pi \int_{[a,b]} h(v) \sqrt{(h'(v)^2+(g'(v))^2}  \diff v = 2\pi \int_C x \diff s
\end{array}
\]
}
