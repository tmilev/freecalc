Integrate 
\begin{enumerate}
\item \label{problemintsecxdx} $\displaystyle \int \sec x  \diff x$.

\answer{ $ \ln |\sec x +\tan x| = \ln \left|\frac{1+\tan \left( \frac{ x}{2}\right)}{1-\tan \left(\frac{x}{2}\right)} \right| +C$}
\item \label{problemintsec^3xdx} $\displaystyle \int \sec^3x  \diff x$.

\answer{$\frac{1}{2}\left( \sec x \tan x +\ln |\sec x +\tan x|\right)+C$}
\item $\displaystyle \int \tan^3x \diff x$.

\answer{$\frac{1}{2}\tan^2x-\ln |\sec x| +C$}
\item $\displaystyle \int \sec^2x\tan^2x \diff x$.

\answer{$\frac{\tan^3 x}{3} +C$}
\end{enumerate}
