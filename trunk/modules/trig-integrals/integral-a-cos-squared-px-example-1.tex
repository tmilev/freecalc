\begin{frame}
\frametitle{Integrals of the form $\int a \cos^2 (px)\diff x$}

\begin{itemize}
\item Integrals of the form 
\[
\int a \alertNoH{2}{\cos^2 (px)}\diff x \uncover<2->{=\int a (\alertNoH{2}{1-\sin^2 (px)})\diff x }
\]
\uncover<2->{can be transformed to a constant minus the already studied integral form $\int \sin^2(px)\diff x $ using $\alertNoH{2}{ \cos^2(px)}=\alertNoH{2}{1-\sin^2(px)}$.}
\item<3-> Alternatively, we show how to solve such integrals directly.
\end{itemize}
\end{frame}
