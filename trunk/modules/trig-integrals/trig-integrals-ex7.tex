\begin{example}[Example 7, p. 500]
\begin{eqnarray*}
\int \tan^3 x \diff x%
 & \uncover<2->{ = } & %
\uncover<2->{\int \tan x \tan^2 x \diff x}\\%
 & \uncover<3->{ = } & %
\uncover<3->{\int \tan x (\sec^2 x - 1)\diff x}\\%
 & \uncover<4->{ = } & %
\uncover<4->{\int \tan x \sec^2 x \diff x - \int \tan x \diff x}\\%
 & \uncover<5->{ = } & %
\uncover<5->{\frac{\tan^2 x}{2}  -  \ln|\sec x|  + C}\\%
\end{eqnarray*}
\end{example}
\end{frame}
