\frametitle{Euler subst. for $\sqrt{x^2+1}$ corresponding to $x=\cot \theta$ }
\begin{itemize}
\item $\alertNoH{4}{ x =\cot \theta}$ transforms $\diff x, x,\sqrt{x^2+1}$ to trig form.
\item $\alertNoH{5}{\theta=2\Arctan t}$, \uncover<3->{ $t>0$} transforms $\diff \theta, \cos\theta,\sin \theta$ to rational form.
\end{itemize}
\uncover<2->{\alertNoH{2}{What if we compose the above?}} \uncover<3->{\alertNoH{3}{We get the Euler substitution:}}
\only<handout:1-4|1-34>{ %
\[
\begin{array}{rclll}
\uncover<3->{\alertNoH{4,16,26,33,34}{x}}
\vphantom{\frac{1}{2}\left(\frac{1}t - t\right)}
&\uncover<3->{\alertNoH{4,16,26,33,34}{=}} &\displaystyle
\vphantom{\frac{1}{2}\left(\frac{1}t -t\right)} %phantom
\only<handout:1|3-13>{
\displaystyle \uncover<4->{ \alertNoH{4}{ \cot \alertNoH{5}{\theta}}} \\
\uncover<5->{ & =& \displaystyle \alertNoH{8}{\cot \left( \alertNoH{ 5}{2\arctan t}\right)}} \uncover<6->{&&| \displaystyle  \text{Recall: } \alertNoH{8}{\cot (2z)} =\frac{ \cos (2z)}{\sin (2z)} \uncover<7->{\alertNoH{8}{=\frac{1-\tan^2z}{2\tan z }}}} \\
\uncover<8->{ &=&\displaystyle \alertNoH{8}{ \frac{1-{\alertNoH{9}{\tan}}^2 \alertNoH{9}{(\arctan t)}}{2 \alertNoH{9}{\tan (\arctan t)}}}} \\

\begin{definition}The Euler substitution for $\sqrt{x^2+1}$ corresponding to $x=\cot \theta$ is given by:
\[
\begin{array}{rcl}
\alertNoH{35}{x} &\alertNoH{35}{=}&\displaystyle \alertNoH{35}{\frac12\left(\frac{1}{t}- t\right) }, \quad \quad t>0\\
\displaystyle \alertNoH{35}{ \sqrt{x^2+1}} & \alertNoH{35}{=}& \displaystyle \alertNoH{35}{\frac12 \left(\frac1t +t\right)} \\
\displaystyle \alertNoH{35}{ \diff x}& \alertNoH{35}{=}&\displaystyle \alertNoH{35}{-\frac12\left(\frac{1}{t^2}+1\right) \diff t}\\
\alertNoH{35}{t} &\alertNoH{35}{=}&\alertNoH{35}{\sqrt{x^2+1}-x}\quad .
\end{array}
\]
\end{definition}
