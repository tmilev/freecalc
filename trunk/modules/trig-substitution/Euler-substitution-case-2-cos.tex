\frametitle{Euler subst. for $\sqrt{-x^2+1}$ corresponding to $x=\cos \theta$ }
\begin{itemize}
\item $\alertNoH{4}{ x =\cos \theta}$ transforms $\diff x, x,\sqrt{-x^2+1}$ to trig form.
\item $\alertNoH{5}{\theta=2\Arctan t}$, \uncover<3->{ $t>0$} transforms $\diff \theta, \cos\theta,\sin \theta$ to rational form.
\end{itemize}
\uncover<2->{\alertNoH{2}{What if we compose the above?}} \uncover<3->{\alertNoH{3}{We get the Euler substitution:}}
\only<handout:1,2,3|1-37>{ %
\[
\begin{array}{rclll}
\uncover<3->{\alertNoH{4,13,22,33,37}{x}}&\uncover<3->{\alertNoH{4,13,22,33,37}{=}}&\displaystyle \vphantom{\frac{1- t^2}{ 1+ t^2} }

\uncover<handout:4|38->{
\begin{definition}
The Euler substitution for $\sqrt{-x^2+1}$ corresponding to $x=\cos \theta$ is given by:
\[
\alertNoH{38}{
\begin{array}{rcl}
x&=&\displaystyle \frac{1-t^2}{1+t^2}, \quad \quad t>0\\
\sqrt{-x^2+1}&=&\displaystyle \frac{2t}{1+t^2}  \\
\diff x&=&\displaystyle  -\frac{4 t}{(t^{2}+1)^{2}} \diff t\\
t&=&\displaystyle \frac{\sqrt{-x^2+1}}{x+1} \quad .
\end{array}
}
\]
\end{definition}
