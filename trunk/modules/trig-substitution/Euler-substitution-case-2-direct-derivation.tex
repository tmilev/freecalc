\begin{frame}
\frametitle{Euler substitution to handle $\sqrt{-x^2+1} $}
\begin{itemize}
\item<1-> Suppose we want to integrate
\[
\int R(x, \sqrt{-x^2+1})\diff x\quad .
\]
\only<2-15>{
\item<2-> Substitute $\alertNoH{9,15}{\sqrt{-x^2+1}}=(1-\alertNoH{11}{x})t  \uncover<11->{ =\left( 1 -\alertNoH{11}{ \left(1-\frac{2}{ t^2+1} \right)} \right)t =\alertNoH{15}{\frac{2t}{t^2+1}}}$.
\item<3-> How is this a substitution of $x$ via t?
\uncover<4->{
\[
\begin{array}{rcll|l}
\sqrt{-x^2+1}&=&(1-x)t&&\text{square}\\
\uncover<5->{(1-x)(1+x)&=&(1-x)^2t^2&&\text{divide by }1-x}\\
\uncover<6->{1+x&=&(1-x)t^2}\\
\uncover<7->{xt^2+x &=&t^2-1}\\
\uncover<8->{x(t^2+1) &=&t^2-1 &&\text{divide by }t^2+1}\\
\uncover<9->{\alertNoH{11,12,15}{x}&\alertNoH{11,15}{=}& \frac{  t^2\uncover<10->{+1-1}-1}{t^2+1} \uncover<10->{ = \alertNoH{11,12,15}{1 -\frac{2}{ t^2+1} }\quad .}}
\end{array}
\]
\item<12->
\[
\alertNoH{12,15}{\diff x =} \alertNoH{12,13,14}{ \diff \left(1-\frac{2}{t^2+1}\right)} \uncover<13->{\alertNoH{13,14,15}{=}}\only<13>{\alertNoH{13}{\textbf{?}}} \uncover<14->{\alertNoH{14,15}{ \frac{4t}{(1+t^2)^2}dt} \quad .}
 {~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~}
\]
}
}
\end{itemize}
\uncover<16->{
\begin{definition}
The Euler substitution for $\sqrt{-x^2+1}$ is the substitution given by:
\[
\begin{array}{rcl}
\displaystyle \alertNoH{16}{\sqrt{-x^2+1}}&\alertNoH{16}{=}& \displaystyle  \alertNoH{16}{\frac{2t}{t^2+1}}\\
\displaystyle  \alertNoH{16}{x}&\alertNoH{16}{=}&\displaystyle  \alertNoH{16}{1-\frac{2}{t^2+1}}\\
\displaystyle \alertNoH{16}{\diff x }&\alertNoH{16}{=} & \displaystyle  \alertNoH{16}{\frac{4t}{(t^2+1)^2}dt }\quad .
\end{array}
\]
\end{definition}
}
