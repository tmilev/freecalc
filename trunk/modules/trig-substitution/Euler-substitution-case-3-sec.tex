\frametitle{Euler substitution $x=\sec \theta$, $\theta = 2\arctan t$}
\begin{itemize}
\item $\alertNoH{4}{ x =\sec \theta}$ transforms $\diff x, x,\sqrt{x^2-1}$ to trig form.
\item $\alertNoH{5}{\theta=2\Arctan t}$, \uncover<3->{ $ t\in (-\infty, -1) \cup \left[0, 1 \right) $ } rationalizes $\diff \theta, \cos\theta,\sin \theta$.
\end{itemize}

\begin{definition}
The Euler substitution for $\sqrt{x^2-1}$ corresponding to $x=\sec \theta$ is given by:
\[
\begin{array}{rcl}
\alertNoH{34}{
x} &\alertNoH{34}{=}&\displaystyle \alertNoH{34}{ \frac{1+t^2}{1-t^2}, \quad \quad \quad t\in (-\infty, -1) \cup \left[0, 1 \right)} \\
\alertNoH{34}{\sqrt{x^2-1}}&\alertNoH{34}{=}&\displaystyle \alertNoH{34}{\frac{2t}{1-t^2}}  \\
\alertNoH{34}{\diff x}&\alertNoH{34}{=}&\displaystyle  \alertNoH{34}{\frac{4 t}{(1- t^{2})^{2}} \diff t}\\
\alertNoH{34}{t}&\alertNoH{34}{=}&\displaystyle \alertNoH{34}{\pm \frac{ \sqrt{x^2-1}}{x+1} }\quad .
\end{array}
\]
\end{definition}
