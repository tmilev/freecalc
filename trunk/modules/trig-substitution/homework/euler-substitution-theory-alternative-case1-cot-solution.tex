\[
\begin{array}{rcll|l}
\sqrt{x^2+1}&=& x+t &&\text{square both sides}  \\
x^2+1&=&x^2+2xt+t^2\\
-2xt&=&t^2-1\\
x&=&\displaystyle\frac{1}{2}\left(\frac{1}{t}-t \right) \quad .
\end{array}
\]
