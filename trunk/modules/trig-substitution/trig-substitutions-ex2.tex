\begin{example} %[Example 2, p. 504]
%\begin{columns}[c]
%\column{.4\textwidth}
Find the area enclosed by the ellipse $\frac{x^2}{a^2} + \frac{y^2}{b^2} = 1$, $a,b>0$.
\begin{columns}
\column{0.37\textwidth}
\psset{xunit=1cm, yunit=1cm}
\begin{pspicture}(-2.500000, -5)(2.500000,5) 
\psframe*[linecolor=white](-2.500000,-5)(2.500000,5) 
\tiny 
\pscustom*[linecolor=\psColorAreaUnderGraph]{
\psplot[linecolor=\psColorGraph, plotpoints=1000]{-2.000000}{2.000000}{1 x 2 exp -0.25 mul add sqrt -1 mul }
\psplot[linecolor=\psColorGraph, plotpoints=1000]{-2.000000}{2.000000}{1 x 2 exp -0.25 mul add sqrt }
}
\psaxesStandard{-2.2}{-1.5}{2.2}{1.5} 
\rput[b](1,1){$y=b\sqrt{1-\frac{x^2}{a^2}} $}
\rput[t](1,-1){$y=-b\sqrt{1-\frac{x^2}{a^2}} $}

%Function formula: - \sqrt{-1/4 x^{2}+1} 
\psplot[linecolor=\psColorGraph, plotpoints=1000]{-2.000000}{2.000000}{1 x 2 exp -0.25 mul add sqrt -1 mul }
%Function formula: \sqrt{-1/4 x^{2}+1} 
\psplot[linecolor=\psColorGraph, plotpoints=1000]{-2.000000}{2.000000}{1 x 2 exp -0.25 mul add sqrt }
\end{pspicture}
The area in question is 
$\begin{array}{l}
\displaystyle \phantom{=}\phantom{4} \int\limits_{\phantom{\textbf{?}}-a}^{\phantom{\textbf{?}}a}2b\sqrt{ 1-\frac{x^2}{ a^2} }\diff x\\ =\displaystyle  4\int\limits_{0}^{a}  b\sqrt{1-\frac{x^2}{a^2}}\diff x\\
=4b\frac{a\pi}{4}=\pi a b
\end{array}
$.
\column{0.63\textwidth} 
\uncover<2->{Use trigonometric substitution: set $x= a\sin \theta $, $\alert<5>{ \theta\in \left(0,\frac{\pi}{2}\right)} $.} \uncover<3->{We have that $\alert<6>{ \sqrt{1-\frac{ {\alert<3>{x}}^2}{a^2}}} =\sqrt{1 -\frac{\alert<4>{ {\alert<3>{a}}^2} {\alert<3>{\sin}}^2\alert<3>{ \theta} }{\alert<4>{ a^2}} } \uncover<4->{ =\sqrt{1- \sin^2\theta}} \uncover<5->{\alert<5,6>{=\cos\theta}}$.} \uncover<7->{When $x=0$, $\theta=0$ and when $x=a$, $\theta=\frac{\pi}{2}$.}

\noindent $
\begin{array}{rcl}
\displaystyle 
\uncover<1->{\int_{0}^{a}\sqrt{1-\frac{x^2}{a^2}}\diff x} &\uncover<6->{\alert<6>{=}}& \displaystyle \int \alert<6>{\cos \theta} ~ \diff \left(a\sin \theta\right)\\
&=&\displaystyle a\int_{0}^a\cos^2 \theta\diff \theta \\
&=&\displaystyle a\int_{0}^{a} \frac{\cos(2\theta)+1}2 \diff \theta\\
&=&\displaystyle a \left[ \frac{\sin (2\theta)}{4} +\frac{\theta}{2}\right]_{\theta=0}^{\theta=\frac{\pi}{2}} \\
&=&a\left(0+\frac{\pi}{4}-(0+0)\right)\\
&=&\frac{a\pi }{4}
\end{array}
$
\end{columns}
\end{example}
