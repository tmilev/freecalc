\begin{example}
\begin{itemize}
\item Find all angles that are coterminal to $\frac{\pi}{4}$.
\item Find all angles in the interval $[-2\pi, \pi]$ that are coterminal to $\frac{\pi}{4}$.
\end{itemize}
\uncover<2->{By theory, the angles coterminal with $\frac{\pi}{4}$ are all angles of the form \[\frac{\pi}{4}+\alertNoH{4}{2k}\pi.\]}


\uncover<3->{ To find which among the angles $\frac{\pi}{4}+\alertNoH{4}{2k}\pi$ lie in the interval $[-2\pi, \pi]$, \alertNoH{4}{we write them as an \alertNoH{4}{infinite} list} \uncover<5->{(\alertNoH{5}{we indicate the unboundedness of the list by ellipsis dots})} \uncover<6->{\alertNoH{6,7}{and cross out the angles that lie outside of the desired interval}.} 
\uncover<4->{\[
\alertNoH{6}{\fcCancel{7}{{\alertNoH{5}{\dots}}}, \fcCancel{7}{\frac{\pi}{4} \alertNoH{4}{-4}\pi}, \alertNoH{8}{ \frac{\pi}{4}\alertNoH{4}{-2}\pi}, \alertNoH{8}{\frac{\pi}{4}}, \fcCancel{7}{ \frac{\pi}{4}+ \alertNoH{4}{2}\pi}, \fcCancel{7}{\frac{\pi}{4}+\alertNoH{4}{4}\pi}, \fcCancel{7}{{\alertNoH{5}{\dots}}}}.
\]
}
\uncover<8->{Our final answer is $\displaystyle \alertNoH{8}{-\frac{7\pi}{4}, \frac{\pi}{4}}$}
}


\end{example}
\end{frame}
