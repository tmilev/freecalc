\begin{enumerate}[ref={\fcProblemRef}]

\answer{$\cos \theta$}
\item $\cos \left(\frac{13\pi}{2}-\theta\right)$.

\answer{$\sin\theta$}
\item $\tan \left(\pi - \theta\right)$

\answer{$-\tan \theta$}
\item $\cot \left(\frac{3\pi}{2} - \theta\right)$

\answer{$\tan \theta $}

\item $\csc \left(\frac{3\pi}{2} + \theta\right)$

\answer{$\sec \theta $}

\end{enumerate}
