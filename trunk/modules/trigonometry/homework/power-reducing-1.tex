Using the power-reducing formulas, rewrite the expression in terms of first powers of the cosines and sines of multiples of the  angle $\theta$.\\


\begin{enumerate}[ref={\fcProblemRef}]
\item $\sin^4\theta$.

\answer{$\frac{1}{8} \cos{}\left(4 \theta\right)-\frac{1}{2} \cos{}\left(2 \theta\right)+\frac{3}{8} $}
\item $\cos^4\theta$.

\answer{$\frac{1}{8} \cos{}\left(4 \theta\right)+\frac{1}{2} \cos{}\left(2 \theta\right)+\frac{3}{8} $}

\item \label{problemExpresssin^6xviaTrigFunctionsMultipleAngle} $\sin^6\theta$.

\answer{$\sin^6\theta=-\frac{1}{32} \cos{}\left(6 \theta\right)+\frac{3}{16} \cos{}\left(4 \theta\right)-\frac{15}{32} \cos{}\left(2 \theta\right)+\frac{5}{16} $}

\item \label{problemExpresscos^6xviaTrigFunctionsMultipleAngle} $\cos^6\theta$.

\answer{$\cos^6\theta=\frac{1}{32} \cos{}\left(6 \theta\right)+\frac{3}{16} \cos{}\left(4 \theta\right)+\frac{15}{32} \cos{}\left(2 \theta\right)+\frac{5}{16}  $}
\end{enumerate}
