Find all solutions of the equation in the interval $[0,2\pi)$. The answer key has not been proofread, use with caution.

\begin{enumerate}[ref={\fcProblemRef}]
\item \label{problemsinx=-sqrt(2)/2} $\sin x = -\frac{\sqrt{2}}{2}$.

\answer{$x=\frac{5\pi}{4}, \frac{7\pi}{4} $}
\item \label{problemcosx=sqrt(3)/2} $\cos x = \frac{\sqrt{3}}{2}$.

\answer{$x= \frac{\pi}{6}, \frac{11\pi}{6}$}
\item \label{problemsin(3x)=1/2} $\sin (3x) = \frac{1}{2}$.

\answer{$x= \frac{\pi}{18},\frac{5\pi}{18}, \frac{13\pi}{18},\frac{17\pi}{18}, \frac{25\pi}{18}, \frac{16\pi}{9}$}
\item \label{problemcos(7x)=0} $\cos (7x) = 0$.

\answer{$x=\frac{\pi}{14}, \frac{3\pi}{14}, \frac{5\pi}{14}, \frac{\pi}{2}, \frac{9\pi}{14}, \frac{11\pi}{14}, \frac{13\pi}{14}, \frac{15\pi}{14}, \frac{17\pi}{14}, \frac{19\pi}{14}, \frac{3\pi}{2}, \frac{23\pi}{14}, \frac{25\pi}{14}, \frac{27\pi}{14} $}
\item \label{problemcos(3x+pi/2)=0} $\cos \left(3x+\frac{\pi}{2}\right) = 0$.

\answer{$x=0, \frac{\pi}{3}, \frac{2\pi}{3}, \pi, \frac{4\pi}{3}, \frac{5\pi}{3} $}
\item \label{problemsin(5x-pi/3)=0}$\sin \left(5x-\frac{\pi}{3}\right) = 0$.

\answer{$x=\frac{\pi}{15}, \frac{4\pi}{15}, \frac{7\pi}{15}, \frac{2\pi}{3}, \frac{13\pi}{15}, \frac{16\pi}{15}, \frac{19\pi}{15}, \frac{22\pi}{15}, \frac{5\pi}{3}, \frac{28\pi}{15}$}
\end{enumerate}
