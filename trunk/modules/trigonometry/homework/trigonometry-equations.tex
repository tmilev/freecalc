\begin{problem}(Textbook page A33, problems 65-72).
Find all values of $x$ in the interval $[0,2\pi]$ that satisfy the 
equation.
\begin{multicols}{3}
\begin{enumerate}
\item $2\cos x - 1=0$.
\item $3\cot^2 x= 1$.
\item $2\sin^2 x= 1$.
\item $|\tan x|=1 $.
\item $\sin 2x = \cos x $.
\item $2\cos x +\sin 2x=0$.
\item $\sin x =\tan x$.
\item $2+\cos 2x = 3 \cos x$.
\end{enumerate}
\end{multicols}
\end{problem}
