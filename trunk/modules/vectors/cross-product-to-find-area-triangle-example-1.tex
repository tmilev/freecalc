\[\begin{array}{rcl}
\displaystyle \text{Area}(ABC) &=&\displaystyle \frac{1}{2}|\fcv{AB} \times \fcv{AC}| =
\frac{1}{2}|( 1,1,-2) \times ( 2, -1, -1) | \\~\\
&=&\displaystyle\frac{1}{2} |( -3, -3, -3 )| \\~\\
&=&\displaystyle \frac{3\sqrt{3}}{2}\; .
\end{array}
