\column{0.6\textwidth}
\begin{itemize}
\item Given: point $P(\fcv{r}_1)=(x_1, y_1, z_1)$,
\item plane $\mathcal{P}: \quad (\fcv{r}-\fcv{r}_0)\cdot \fcv{n} = 0$.
\item Goal: find the distance between $P$ and $\mathcal P$.
\end{itemize}
\end{columns}
\alertNoH{1-}{Distance} from $P$ to $\mathcal{P}$:
\uncover<2->{
$\begin{array}{rcl}
d(P,\mathcal{P}) &\alertNoH{0}{=}&\displaystyle |\fcv{proj}_{\fcv{n}}( \fcv{r}_1-\fcv{r}_0)|\\
\alertNoH{7}{d(P ,\mathcal{P})} &\alertNoH{7}{=}&\displaystyle \alertNoH{7}{ \frac{|(\fcv{r}_1- \fcv{r }_0) \cdot \fcv{n}|}{|\fcv{n}|}}
\end{array}
$
}

\uncover<3->{\alertNoH{1-}{Scalar equation}:
$
\begin{array}{rrcl}
\mathcal{P}:& \alertNoH{4}{a} \alertNoH{5}{x}+ \alertNoH{4}{b}\alertNoH{5}{y}+ \alertNoH{4}{c} \alertNoH{5}{z} +\alertNoH{6,9}{ d} = (\alertNoH{5}{\fcv r} \alertNoH{6,9}{- \fcv r_0})\alertNoH{6,9}{\cdot\alertNoH{4}{\fcv n}}&=&0\\
\uncover<4->{
&\alertNoH{4}{\fcv{n}} &=& \alertNoH{4}{( a,b,c)} }\\
\uncover<7->{&\displaystyle \alertNoH{7}{ d(P, \mathcal P)= \frac{|( \alertNoH{8}{ \fcv{r}_1} \alertNoH{9}{-\fcv{r}_0}) \alertNoH{8,9}{\cdot \fcv{n}}|}{\alertNoH{10}{|\fcv{n}|}}} & =&\displaystyle \frac{|\alertNoH{8}{ax_1+by_1+cz_1} +\alertNoH{9}{d} |}{\alertNoH{10}{ \sqrt{a^2+b^2+c^2}}}}
\end{array}$
}
