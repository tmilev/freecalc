Write vectorial and scalar equations of the line passing $L$ through the given points.

\begin{enumerate}
\item $(2,3,5)$ and $(3,5,7)$.

\answer{
$L: ( 2, 3, 5 ) + t( 1, 2, 2 )
$
or
$L:\left|\begin{array}{rcl}
x&=& 2+t\\
y&=& 3+2t\\
z&=& 5+2t
\end{array}\right., t\in \mathbb R
$
}

\item $(-1,-1,1)$ and $(-1,1,-1)$.

\answer{
$L: ( -1, 0,0 ) + t( 0,1,-1 )
$
or
$L:\left|\begin{array}{rcl}
x&=& -1\\
y&=& t\\
z&=& -t
\end{array}\right., t\in \mathbb R
$
}
\end{enumerate}
