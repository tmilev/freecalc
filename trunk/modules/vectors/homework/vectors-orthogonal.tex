Determine if the vectors are orthogonal.

\begin{enumerate}
\item $\fcv u=(1, 2, 3 )$, $\fcv v=(-1,2,-1)$.
\answer{$\fcv u\perp \fcv v$}
\item $\fcv u=( 1, 0, 1 )$, $\fcv v=(-1,1,1)$.
\answer{$\fcv u\perp \fcv v$}
\item $\fcv u=( -1, 0, 1 )$, $\fcv v=(-1,1,1)$.
\answer{$\fcv u\not\perp \fcv v$}
\end{enumerate}
