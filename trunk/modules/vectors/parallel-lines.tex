\begin{itemize}
\item Given: lines
    $$L_1: \quad \fcv{r}= \fcv{r}_1+t\fcv{u}_1 \qquad L_2: \quad \fcv{r}= \fcv{r}_2+s\fcv{u}_2$$
\item Goal: distance between lines.
\end{itemize}
\alert<1->{Parallel} lines
\uncover<2->{
$L_1 || L_2$ $\Longleftrightarrow$
$\fcv{u}_1$, $\fcv{u}_2$ collinear $\Longleftrightarrow$
$\boxed{\fcv{u}_1 \times \fcv{u}_2 = \fcv{0}}$
}
\uncover<3->{
\alert<1->{Distance}:
$$d= d(L_1,L_2)  = d(P_1,L_2) = d(P_2,L_1)$$}
\uncover<4->{
$$d= d(L_1,L_2) = |\fcv{\text{orth}}_{\bm{u}_1} (\fcv{r}_2-\fcv{r}_1)|$$
$$\boxed{d = \frac{|(\fcv{r}_2-\fcv{r}_1) \times \fcv{u}_1|}{|\fcv{u}_1|} =\frac{|(\fcv{r}_2-\fcv{r}_1) \times \fcv{u}_2|}{|\fcv{u}_2|}}$$}
    %\textcolor[rgb]{0.98,0.00,0.00}{Identical} lines: $d(L_1,L_2)=0$
    %$$\fcv{u}_1\times \fcv{u}_2 = \fcv{0} \text{ and }
    %(\fcv{r}_2-\fcv{r}_1) \times \fcv{u}_1 = \fcv{0}$$
