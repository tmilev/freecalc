\begin{example}
Find equations of a plane passing through $P_0\alertNoH{7}{(1,2,3)}$ and parallel to the vectors $\fcv{u}=\alertNoH{3}{( -1,0,2)}$, $\fcv{v} = \alertNoH{3}{( 0,-2,1)}$.
\[
\uncover<2->{\alertNoH{2,6}{\fcv{n} =}}
\uncover<3->{
\fcv{u} \times \fcv{v}
=\left|
\begin{array}{ccc}
\fcv{i} & \fcv{j} & \fcv{k} \\
\alertNoH{3}{-1} & \alertNoH{3}{0} & \alertNoH{3}{2} \\
\alertNoH{3}{0} & \alertNoH{3}{-2} & \alertNoH{3}{1}
\end{array}
\right|
= \alertNoH{6}{4\fcv{i}+\fcv{j} +2\fcv{k}}
}%uncover and alert
\]
\uncover<4->{\alertNoH{4}{$\Rightarrow$ implicit scalar equation given by: }}
\uncover<5->{\alertNoH{5}{$\alertNoH{6}{4}(x-\alertNoH{7}{1})+\alertNoH{6}{1}(y-\alertNoH{7}{2}) + \alertNoH{6}{2}(z-\alertNoH{7}{3}) = 0 \uncover<8->{\Longleftrightarrow 4x+y+2z = 12}$}}%

\uncover<9->{\alertNoH{9,10}{Parametric vectorial equation: }}
\uncover<10->{\alertNoH{10}{$( x, y, z ) = ( 1,2,3) + s( -1, 0, 2) + t( 0,-2,1)$
}}

\uncover<11->{\alertNoH{11,12}{Parametric scalar equations:}}
\uncover<12->{\alertNoH{12}{
$$\left| \begin{array}{ll}
x & = 1 -s \\
y & = 2-2t \\
z & = 3 +2s +t
\end{array}
\right. \quad s,t \text{ real parameters}.$$
}}
\end{example}
