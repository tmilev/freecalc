\begin{itemize}
\item Given: point $P_0$ with position vector $\fcv{r}_0$. \item<2-> Non-parallel directions $\fcv{u}$ and $\fcv{v}$.
\item<3-> Goal: give equations of plane $\mathcal P$ through $P_0$ and parallel to both $\fcv{u}$ and $\fcv{v}$.
\end{itemize}
\end{columns}
\only<handout:1|4-6>{%
\uncover<4->{Normal direction $\fcv{n} = \fcv{u} \times \fcv{v} \neq \fcv{0}$. }

\uncover<5->{\alertNoH{1-}{Implicit equation}: $P(\fcv{r}) \text{ is on } \mathcal{P} \Longleftrightarrow (\fcv{r}-\fcv{r}_0) \cdot \fcv{n} = 0$}
\uncover<6->{Interpretation:$$\text{Vol}(R(\fcv{r} - \fcv{r}_0,\fcv{u},\fcv{v})) = 0$$}
} %uncover

\only<handout:1|7-10>{$P(\fcv{r})$ is on the plane $\mathcal{P}$ $\Leftrightarrow$

\uncover<8->{ $\fcv{P}_0\fcv{P}$ is a combination of $\fcv{u}$, $\fcv{v}$ $\Leftrightarrow$
}

\uncover<9->{There are scalars $s$, $t$ such that $\fcv{r}-\fcv{r}_0= s\fcv{u} + t\fcv{v} \Leftrightarrow $}

\uncover<handout:2|10->{%
\alertNoH{1-}{Parametric equation}:
\[\alertNoH{10}{ \fcv{r} = \fcv{r}_0 + s\fcv{u} + t\fcv{v}}
\]
for some parameters $s$ and $t$.
}%uncover
}%only<7-?>
\only<handout:2|11->{

\[
\alertNoH{11}{ \alertNoH{13}{\fcv{r}} = \alertNoH{14}{\fcv{r}_0} + s \alertNoH{15}{\fcv{u}} + t\alertNoH{16}{\fcv{v}}}
\]

\uncover<12->{Let $P_0(x_0,y_0,z_0)$, $P(x,y,z)$ $\fcv{u} = ( u_1,u_2,u_3)$, $\fcv{v}=( v_1,v_2,v_3)$.} \uncover<13->{$\Rightarrow$}

$$
